2/**
3 * TextPosterGenerator — renders a text-only submission as a placeholder poster.
4 *
5 * pure@ (agnosis_artwork, zero AI, the artist's own words verbatim) is a
6 * visual-art lane — an agnosis_artwork post is meant to show something. Once
7 * Email\Parser stopped requiring an attachment (poetry and other text-only
8 * submissions are art too — see Parser::parse_imap_message()'s relaxed
9 * gate), a pure@ submission with no photo would otherwise publish with an
10 * empty gallery. This class generates a stand-in image from the artist's own
11 * text instead, so the gallery still shows something they actually wrote.
12 *
13 * Design — "edge-to-edge overflow poster" (confirmed with the site owner):
14 * each line of the submission's own text is rendered at the largest font
15 * size that makes THAT line span the canvas edge-to-edge, then all lines are
16 * stacked with their vertical centers evenly distributed across the canvas
17 * height. This means:
18 *   - A short submission (a few lines) reads cleanly, generously spaced —
19 *     each line big, bold, on its own.
20 *   - A long submission (many lines) degrades gracefully into overlap — the
21 *     later lines increasingly stack on top of each other, so only isolated
22 *     words or short phrases stay legible near the top. The concealment is
23 *     purely a side effect of scale and layout, never content omission: no
24 *     text is redacted or truncated (beyond the sane MAX_LINES processing
25 *     cap below), it simply becomes visually dense.
26 *
27 * Styling matches the site's own dark brand (agnosis-theme's theme.json
28 * palette: background #0d0d12, foreground #ededf0) and bundles its own copy
29 * of Plus Jakarta Sans (SIL OFL 1.1 — see assets/fonts/plus-jakarta-sans/
30 * LICENSE.txt) rather than reading the theme repo's font files at runtime —
31 * a plugin's core functionality should never depend on a specific theme
32 * being active.
33 *
34 * Imagick is a soft dependency everywhere else in this codebase (see
35 * AI\MediaAdapter) — generate() follows the same extension_loaded() guard
36 * and graceful-degradation convention: on any failure it returns null and
37 * logs a reason, and the caller (PostCreator::handle()) publishes text-only
38 * rather than blocking the submission.
39 *
40 * @package Agnosis\Publishing
41 */

152    /**
153     * Split the submission into non-empty, trimmed lines, preferring the
154     * body text and falling back to the subject only when the body is
155     * entirely empty (a title-only submission).
156     *
157     * Preserves the artist's own line breaks rather than word-wrapping —
158     * for a poem, the line structure is part of the work, not just filler.
159     *

162    private static function extract_lines( string $subject, string $text ): array {

164        if ( '' === trim( $source ) ) {
165            return [];

169        $lines     = array_values( array_filter(

177    /**
178     * Find the font size that makes $line span roughly $usable_width px —
179     * measured once at a reference size and scaled linearly (text width
180     * scales ~linearly with font size for a fixed string), then clamped so
181     * a single short word doesn't balloon to an absurd size and a
182     * many-word line doesn't shrink into illegibility.

184    private static function fit_font_size( \Imagick $canvas, string $line, int $usable_width, string $font ): int {
185        $reference = new \ImagickDraw();
186        $reference->setFont( $font );
187        $reference->setFontSize( 100 );

189        $metrics = $canvas->queryFontMetrics( $reference, $line );
190        $measured_width = max( 1.0, (float) $metrics['textWidth'] );
191
192        $scaled = (int) round( 100 * ( $usable_width / $measured_width ) );
193
194        return max( self::MIN_FONT_SIZE, min( self::MAX_FONT_SIZE, $scaled ) );
