2/**
3 * Stateless unsubscribe tokens for the artist newsletter.
4 *
5 * Admitted artists are auto-enrolled in the artist newsletter (no signup row
6 * to key off of), so their one-click unsubscribe link uses a stateless HMAC
7 * token — the same pattern as VouchConfirm's vote links — instead of a stored
8 * per-user secret. Nothing needs to be provisioned ahead of time and the
9 * token is cheap to verify.
10 *
11 * @package Agnosis\Newsletter
12 */
13
14declare(strict_types=1);
15
16namespace Agnosis\Newsletter;
17
18class Tokens {
19
20    /** Build the artist newsletter unsubscribe token for a given user. */
21    public static function artist_unsubscribe_token( int $user_id ): string {
22        return hash_hmac( 'sha256', 'newsletter-optout|' . $user_id, wp_salt( 'auth' ) );
23    }
24
25    /** Constant-time verification of an artist unsubscribe token. */
26    public static function verify_artist_unsubscribe_token( int $user_id, string $token ): bool {
27        return hash_equals( self::artist_unsubscribe_token( $user_id ), $token );
28    }
