2/**
3 * Self-hosted update checker.
4 *
5 * Hooks into WordPress's plugin-update machinery so wp-admin surfaces
6 * "Update available" notices and runs the one-click updater — without
7 * Agnosis being listed on WordPress.org (it isn't; see the self-hosted
8 * `load_plugin_textdomain()` note in Plugin::load_textdomain()).
9 *
10 * Flow:
11 *  1. On every WordPress update check (`pre_set_site_transient_update_plugins`),
12 *     fetch a small JSON manifest from agnosis.art and cache it for
13 *     CACHE_TTL seconds (default: 12 h).
14 *  2. If the manifest version is newer than the installed version, inject an
15 *     entry into `$transient->response`, which is what WordPress reads to show
16 *     the update badge and trigger the one-click updater.
17 *  3. On the `plugins_api` hook, respond to "plugin_information" requests for
18 *     our slug, returning a populated object that fills the "View version
19 *     details" modal (changelog, description, etc.).
20 *  4. After a successful update, purge the cached manifest so the next check
21 *     re-fetches immediately.
22 *
23 * Manifest format — see docs/agnosis-update-manifest.php for the deployable
24 * template (mu-plugin on agnosis.art) and its schema.
25 *
26 * Modeled directly on Lingua Forge's `Linguaforge_Updater` (the companion
27 * plugin's `includes/class-updater.php`) so both self-hosted plugins behave
28 * identically for operators running either or both.
29 *
30 * @package Agnosis\Core
31 */
32

115        if ( empty( $transient->checked ) ) {
116            return $transient;
117        }

120        if ( ! $manifest ) {

122        }
123

125        // from the AGNOSIS_VERSION constant. When WordPress re-runs this filter
126        // immediately after an upgrade (still within the same update.php
127        // request), the constant still reflects the old version while the file
128        // on disk already contains the new one. Reading from disk prevents us

142            $transient->no_update[ AGNOSIS_BASENAME ] = $this->build_no_update_object( $manifest );

278    /**
279     * Hook: upgrader_pre_download
280     *
281     * Intercepts the download of our own update package to:
282     *  1. Pin the download host — a manipulated manifest cannot redirect the
283     *     upgrade to an arbitrary server.
284     *  2. Verify the SHA-256 hash declared in the manifest (when the field is
285     *     present). The check is skipped gracefully when the manifest omits
286     *     `sha256` so that existing releases without the field still update.
287     *
288     * We download the ZIP ourselves, verify it, and return the local temp-file
289     * path. WordPress then uses that path directly instead of re-downloading,
290     * so the file is only fetched once.
291     *
292     * Returns false for any package that is not ours, letting WordPress handle
293     * it normally. Returns a WP_Error to abort the update on a failed check.
294     *
295     * @param false|string|\WP_Error $pre       Pre-existing result (false = not handled yet).
296     * @param string                 $package   Download URL passed to the upgrader.
297     * @param \WP_Upgrader           $_upgrader The upgrader instance (unused).
298     * @return false|string|\WP_Error
299     */
300    public function verify_and_download( $pre, string $package, \WP_Upgrader $_upgrader ) { // phpcs:ignore Generic.CodeAnalysis.UnusedFunctionParameter.FoundAfterLastUsed -- $_upgrader required by upgrader_pre_download's filter signature.
301        // If a prior filter already handled this, respect it.
302        if ( false !== $pre ) {
303            return $pre;
304        }
305
306        $manifest = $this->fetch_manifest();
307        if ( ! $manifest || empty( $manifest->download_url ) ) {
308            return false;
309        }
310
311        // Only intercept our own package URL.
312        if ( $package !== $manifest->download_url ) {
313            return false;
314        }
315
316        // Host pinning — block if the download URL resolves to an unexpected host.
317        $host = wp_parse_url( $package, PHP_URL_HOST );
318        if ( ! $this->is_allowed_download_host( (string) $host ) ) {
319            return new \WP_Error(
320                'agnosis_updater_host_blocked',
321                sprintf(
322                    /* translators: %s: download URL hostname */
323                    __( 'Agnosis update blocked: download host "%s" is not on the allowlist.', 'agnosis' ),
324                    (string) $host
325                )
326            );
327        }
328
329        // Download to a temp file.
330        $tmp = download_url( $package );
331        if ( is_wp_error( $tmp ) ) {
332            return $tmp;
333        }
334
335        // SHA-256 verification — skipped when the manifest omits the field so
336        // existing releases without a hash still update cleanly.
337        if ( ! empty( $manifest->sha256 ) ) {
338            $actual   = hash_file( 'sha256', $tmp );
339            $expected = strtolower( trim( (string) $manifest->sha256 ) );
340
341            if ( ! hash_equals( $expected, (string) $actual ) ) {
342                wp_delete_file( $tmp );
343                return new \WP_Error(
344                    'agnosis_updater_checksum_mismatch',
345                    __( 'Agnosis update blocked: SHA-256 of the downloaded package does not match the manifest. The file may have been tampered with.', 'agnosis' )
346                );
347            }
348        }
349
350        return $tmp;
351    }

353    /**
354     * Returns true when the given hostname is on the download allowlist.
355     *
356     * Exact matches and subdomains of allowlisted hosts are both accepted
357     * (e.g. "releases.github.com" matches "github.com").
358     *

363        foreach ( self::ALLOWED_DOWNLOAD_HOSTS as $allowed ) {
364            if ( $host === $allowed || str_ends_with( $host, '.' . $allowed ) ) {
365                return true;
