640                continue; // Old option doesn't exist — nothing to migrate.

643            if ( false === get_option( $new_key, false ) ) {

645                // verbatim, including an intentionally-blank old value.
646                update_option( $new_key, get_option( $old_key, '' ) );
647            }

649            delete_option( $old_key );
650        }
