334            $msg = trim( $e->getMessage() );
335            // Deduplicate adjacent identical messages.
336            if ( '' !== $msg && ! in_array( $msg, $seen, true ) ) {
337                $parts[] = $msg;
338                $seen[]  = $msg;
339            }
340            $e = $e->getPrevious();
341        }
342        return implode( ' → ', $parts );
343    }

351    /**
352     * Build and connect a webklex IMAP Client.
353     *
354     * webklex/php-imap's native 'imap' protocol driver connects via
355     * stream_socket_client("ssl://host:port"), which PHP automatically uses to
356     * send the SNI hostname in the TLS handshake. This allows Dovecot on a
357     * shared IP to present the correct domain certificate — something the
358     * built-in PHP imap_open() (c-client) cannot do because c-client has no
359     * SNI support.
360     *
361     * We go through ClientManager::make() rather than instantiating Client
362     * directly. The Client constructor's setAccountConfig() requires a non-null
363     * default config which only ClientManager provides.
364     *
365     * @throws \Webklex\PHPIMAP\Exceptions\ConnectionFailedException When the TCP/SSL connection cannot be established.
366     * @throws \Webklex\PHPIMAP\Exceptions\AuthFailedException When the IMAP LOGIN command is rejected.
367     */
368    private function make_client(): Client {
369        $host    = (string) get_option( 'agnosis_imap_host' );
370        $port    = (int) get_option( 'agnosis_imap_port', 993 );
371        $ssl     = (bool) get_option( 'agnosis_imap_ssl', true );
372        $no_cert = (bool) get_option( 'agnosis_imap_novalidate_cert', false );
373        $user    = (string) get_option( 'agnosis_imap_user' );
374        $pass    = (string) get_option( 'agnosis_imap_pass' );
375
376        // ClientManager is required — instantiating Client directly passes null
377        // as the default_config argument, triggering a TypeError in v5.
378        $manager = new ClientManager( [] );
379
380        $client = $manager->make( [
381            'host'          => $host,
382            'port'          => $port,
383            // 'imap' uses the native PHP stream driver (SNI-capable).
384            // Never use 'legacy-imap' — that falls back to imap_open() which has no SNI.
385            'protocol'      => 'imap',
386            'encryption'    => $ssl ? 'ssl' : false,
387            'validate_cert' => ! $no_cert,
388            'username'      => $user,
389            'password'      => $pass,
390        ] );
391
392        $client->connect();
393        return $client;
394    }

396    /**
397     * Detect an IMAP UIDVALIDITY change and reset the UID cursor when one
398     * occurs (reliability audit §5d).
399     *
400     * Per RFC 3501 §2.3.1.1, a UID is only guaranteed stable within one
401     * UIDVALIDITY generation for a folder — a mailbox rebuild, a provider
402     * migration, or the folder itself being recreated can (and in practice
403     * does) make the server reissue UIDs starting from 1 again while
404     * UIDVALIDITY changes to signal exactly that. Without this check,
405     * `agnosis_imap_last_uid` survives such an event unnoticed: every new
406     * message now has a UID below the stale cursor, `UID N+1:*`
407     * (query_messages()) matches nothing ever again, and the failure is
408     * silent — no error, the cron keeps running, the queue keeps "draining"
409     * (there's simply nothing new in it) — indistinguishable from the
410     * outside from "artists just stopped emailing."
411     *
412     * Called at the very top of process_messages(), before the cursor is
413     * read, so a reset here lands in the very same poll: query_messages()
414     * then takes its existing $last_uid === 0 branch and rescans the
415     * date-bounded retention window instead of an empty `UID 1:*`.
416     *
417     * @param Folder $folder The currently selected/examined IMAP folder.
418     */
419    private function check_uidvalidity( Folder $folder ): void {
420        try {
421            $status = $folder->getStatus();
422        } catch ( \Throwable $e ) {
423            // This is a diagnostic add-on to the cursor, never a prerequisite
424            // for polling — if the server doesn't answer STATUS cleanly,
425            // fall through and let the existing cursor logic run exactly as
426            // it did before this check existed, rather than failing the poll
427            // over a check whose whole purpose is preventing silent failure.
428            Logger::warning( 'Poll: could not read folder STATUS for the UIDVALIDITY check — ' . $e->getMessage(), 'inbox' );
429            return;
430        }
431
432        $current = (int) ( $status['uidvalidity'] ?? 0 );
433        if ( $current <= 0 ) {
434            return; // Server didn't report one (rare, but MUST is not MUST-implement-correctly) — nothing to compare.
435        }
436
437        $stored = (int) get_option( 'agnosis_imap_uidvalidity', 0 );
438
439        if ( 0 === $stored ) {
440            // First poll we've ever tracked UIDVALIDITY for this mailbox —
441            // nothing to compare against yet, just start tracking it.
442            update_option( 'agnosis_imap_uidvalidity', $current );
443            return;
444        }
445
446        if ( $stored !== $current ) {
447            Logger::warning(
448                sprintf(
449                    'Poll: IMAP UIDVALIDITY changed (%d → %d) — the mailbox was rebuilt, migrated, or the folder recreated. Resetting the UID cursor to 0 so the next query rescans the retention window instead of silently matching nothing.',
450                    $stored,
451                    $current
452                ),
453                'inbox'
454            );
455            update_option( 'agnosis_imap_last_uid', 0 );
456            update_option( 'agnosis_imap_uidvalidity', $current );
457        }
458    }
459
460    /**
461     * Fetch recent messages and enqueue any not yet in the pipeline.
462     *
463     * We query ALL messages within the retention window rather than only UNSEEN
464     * ones. This avoids a race condition where a concurrent mail client (e.g.
465     * Apple Mail) re-marks messages as \Seen between our clearFlag and our query,
466     * causing the UNSEEN search to return 0.  The database `is_already_queued()`
467     * state machine is the authoritative idempotency guard — IMAP flags are not.
468     *
469     * @param Folder $folder
470     */
471    private function process_messages( Folder $folder ): void {
472        // UIDVALIDITY guard (security/reliability audit §5d) — must run
473        // before the cursor below is read, so a reset here is picked up by
474        // this same poll cycle rather than the next one.
475        $this->check_uidvalidity( $folder );
476
477        // ---- Build the query — headers-first, UID-sequenced -----------------
478        //
479        // fetchBody(false): webklex fetches only RFC-822 headers on the first
480        // IMAP FETCH.  The body is pulled lazily when parse_imap_message() is
481        // called below — and only for messages that clear all cheap gates.
482        //
483        // setSequence(ST_UID): the IMAP server numbers messages two ways:
484        //   • Message Sequence Numbers (MSN) — renumber when older mail is deleted.
485        //   • UIDs — stable for the lifetime of a mailbox.
486        // Operating in UID mode means our cursor survives expunges and reconnects.
487        //
488        // UID cursor: on every run after the first we send `UID N+1:*` which tells
489        // the server to return only messages whose UID is strictly greater than the
490        // last one we processed.  First run (or cursor reset) falls back to the
491        // configured retention window so we do not skip legitimate back-dated mail.
492
493        $last_uid = max( 0, (int) get_option( 'agnosis_imap_last_uid', 0 ) );
494        $messages = $this->query_messages( $folder, $last_uid );
495        $max_uid_seen = $last_uid;
496
497        $goodbye_addr   = strtolower( trim( (string) get_option( 'agnosis_email_goodbye', '' ) ) );
498        $community_addr = strtolower( trim( (string) get_option( 'agnosis_email_community', '' ) ) );
499
500        foreach ( $messages as $message ) {
501            try {
502                $uid = (string) $message->getUid();
503
504                // Track the highest UID we have seen in this poll so we can
505                // advance the cursor even for messages we decide to skip.
506                $numeric_uid = (int) $uid;
507                if ( $numeric_uid > $max_uid_seen ) {
508                    $max_uid_seen = $numeric_uid;
509                }
510
511                if ( $this->is_already_queued( $uid ) ) {
512                    continue; // DB state machine handles all cases.
513                }
514
515                // ---- Cheap header checks (before body download) --------------------
516                // Read From: and To: from headers — no body fetch needed. These are
517                // fast IMAP FETCH (RFC 822 header) operations. All expensive work
518                // (body download, attachment parsing, AI) only runs for admitted senders
519                // that also pass throttle and auth checks.
520
521                $from_list  = $message->getFrom()->toArray();
522                $from_email = $from_list ? sanitize_email( (string) $from_list[0]->mail ) : '';
523
524                // Resolved once, up front (fourth audit §3b): the auth gate just below
525                // — now evaluated before any alias routing — needs an artist_id for
526                // accurate logging/admin-table display, the same identity "cheap gate
527                // 1" further down used to resolve when auth was checked after it. This
528                // is a plain lookup for logging purposes only, NOT an admission check —
529                // the goodbye/community alias handlers below do their own, and gate 1
530                // below still performs the real admitted-artist gate.
531                $artist_user = $from_email ? get_user_by( 'email', $from_email ) : null;
532                $artist_id   = $artist_user ? $artist_user->ID : null;
533
534                // --- Bounce/complaint DSN (security audit §5a) — checked BEFORE the
535                // auth gate below, deliberately: a DSN is a legitimate delivery-status
536                // notification from the RECEIVING mail server, not a message an artist
537                // authenticated, so it will routinely fail SPF/DKIM-for-artist-domain
538                // checks when agnosis_require_email_auth is on. Requiring that gate
539                // here would silently discard the bounce signal that gate has nothing
540                // to do with. Previously (and still, when this doesn't match) every DSN
541                // fell through to "cheap gate 1" below and was recorded as an
542                // unregistered_sender skip — noise in the Inbox table that also threw
543                // away the one thing worth reading in the message.
544                if ( $this->is_bounce_dsn( $message ) ) {
545                    $this->handle_bounce_dsn( $message, $from_email, $uid );
546                    continue;
547                }
548
549                // --- Auth gate (opt-in, fourth audit §3b) — evaluated BEFORE any
550                // alias routing. Previously this ran as "cheap gate 3", ~40 lines
551                // below, reached only by messages that fell through to the normal
552                // artwork/bio/event pipeline — so the opt-in SPF/DKIM hardening this
553                // option promises never covered the two aliases where a spoofed From
554                // does the most damage: an impersonated community@ broadcast (relayed
555                // to every other artist under the spoofed sender's own name) and a
556                // forged goodbye@ removal request. Moving the check here closes both.
557                if ( ! $this->passes_email_auth( $message, $from_email, $uid ) ) {
558                    $this->mark_no_artwork( $uid, $artist_id, 'auth_failed', $from_email );
559                    continue;
560                }
561
562                // --- Goodbye alias: self-removal request (no attachment required) ---
563                // fifth audit §5a: matched against every To:/Cc: recipient, not
564                // just To[0] — an artist writing to goodbye@ while CCing someone
565                // else, or whose mail client serialised To: in an unexpected
566                // order, used to fall through to the normal pipeline instead.
567                if ( $goodbye_addr ) {
568                    $recipients = $this->message_recipient_addresses( $message );
569                    if ( in_array( $goodbye_addr, $recipients, true ) ) {
570                        $this->handle_goodbye_email( $from_email, $uid );
571                        continue;
572                    }
573                }
574
575                // --- Community alias: broadcast to all other artists (no attachment required) ---
576                if ( $community_addr ) {
577                    $recipients = $this->message_recipient_addresses( $message );
578                    if ( in_array( $community_addr, $recipients, true ) ) {
579                        $this->handle_community_email( $message, $from_email, $uid );
580                        continue;
581                    }
582                }
583
584                // --- Cheap gate 1: admitted sender (header-only) -------------------
585                // $artist_user / $artist_id already resolved above.
586                if ( null === $artist_id ) {
587                    Logger::warning( 'Skipped UID ' . $uid . ': unregistered sender <' . $from_email . '>.', 'inbox' );
588                    $this->mark_no_artwork( $uid, null, 'unregistered_sender', $from_email );
589                    continue;
590                }
591
592                if ( ! $this->is_admitted_artist( $artist_id ) ) {
593                    Logger::warning( 'Skipped UID ' . $uid . ': sender <' . $from_email . '> (user #' . $artist_id . ') is not admitted.', 'inbox' );
594                    $this->mark_no_artwork( $uid, $artist_id, 'not_admitted', $from_email );
595                    continue;
596                }
597
598                // --- Cheap gate 2: per-sender intake throttle ----------------------
599                $sender_limit  = max( 1, (int) get_option( 'agnosis_intake_per_sender_limit', 5 ) );
600                $throttle      = RateLimiter::check_sender( 'email_intake', $from_email, $sender_limit, HOUR_IN_SECONDS );
601                if ( is_wp_error( $throttle ) ) {
602                    Logger::warning( 'Skipped UID ' . $uid . ': sender <' . $from_email . '> throttled (' . $sender_limit . '/hour limit).', 'inbox' );
603                    $this->mark_no_artwork( $uid, $artist_id, 'throttled', $from_email );
604                    continue;
605                }
606
607                // Cheap gate 3 (SPF/DKIM) used to run here — moved above alias
608                // routing (fourth audit §3b); see passes_email_auth() and its call
609                // site near the top of this loop, right after $from_email/$artist_id
610                // are resolved. Every message reaching this point has already passed
611                // it, regardless of which branch (goodbye/community/normal) it took.
612
613                // ---- Full parse (body + attachments) — admitted senders only ------
614                $submission = $this->parser->parse_imap_message( $message );
615
616                if ( null === $submission ) {
617                    // No valid image attachments — mark it so we don't re-check it. The
618                    // sender IS a known, admitted artist here (both gates above already
619                    // passed) — persist $artist_id so the Inbox admin table shows who
620                    // this was instead of "unregistered sender", and so a later
621                    // auto-retry (is_already_queued()'s 'failed' branch) doesn't have
622                    // PostCreator wrongly re-reject a real artist as unadmitted.
623                    Logger::info( 'Skipped UID ' . $uid . ': no valid image attachments.', 'inbox' );
624
625                    if ( Debug::enabled() ) {
626                        // Parser::parse_imap_message() already wrote a detailed
627                        // 'parser-attachments-*' dump (raw MIME structure, per-
628                        // attachment MIME sniff/declared/disposition/size, and
629                        // why each one was accepted or rejected) for this exact
630                        // UID just above, in the call this branch is reacting
631                        // to. This companion file just makes that dump easy to
632                        // find from a directory listing sorted by time — the
633                        // two should have matching timestamps.
634                        Debug::write(
635                            'inbox-skip-no-attachments',
636                            sprintf(
637                                "UID: %s\nFrom: %s\nSubject: %s\nArtist user ID: %d\n\nSee the parser-attachments-* dump written immediately before this one (same poll cycle) for the full MIME/attachment trace.",
638                                $uid,
639                                $from_email,
640                                (string) $message->getSubject(),
641                                $artist_id
642                            )
643                        );
644                    }
645
646                    $this->mark_no_artwork( $uid, $artist_id, 'no_attachments', $from_email );
647
648                    /**
649                     * Fires when an admitted artist's email was received but contained no
650                     * recognizable image/audio/video attachment (e.g. a photo inserted
651                     * inline rather than properly attached).
652                     *
653                     * @param int    $artist_id WordPress user ID of the sender.
654                     * @param string $uid       IMAP message UID (for logging/reference only).
655                     */
656                    do_action( 'agnosis_submission_no_attachment', $artist_id, $uid );
657                    continue;
658                }
659
660                $this->enqueue( $uid, $submission );
661
662            } catch ( \Throwable $e ) {
663                Logger::error( 'Error processing message UID ' . ( $uid ?? '?' ) . ': ' . $e->getMessage(), 'inbox' );
664            }
665        }
666
667        // Advance the UID cursor so the next poll only fetches genuinely new mail.
668        if ( $max_uid_seen > $last_uid ) {
669            update_option( 'agnosis_imap_last_uid', $max_uid_seen );
670            Logger::info( 'Poll: UID cursor advanced to ' . $max_uid_seen . '.', 'inbox' );
671        }
672    }
673
674    /**
