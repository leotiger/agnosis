254    /**
255     * Whether a webhook payload carries an `Auto-Submitted` header indicating
256     * it is an automated response (vacation auto-reply, mailing-list bounce,
257     * etc.) rather than a human-written email (fourth audit §3c).
258     *
259     * Per RFC 3834, genuine auto-responses set this header to a value other
260     * than `no` (typically `auto-replied`) — normal mail either omits the
261     * header or, less commonly, sets it explicitly to `no`; only a present,
262     * non-`no` value trips this guard, so an absent header (the overwhelming
263     * majority of real mail) is not treated as suspicious.
264     *
265     * @param array<string, mixed> $payload Raw webhook POST params.
266     * @return bool True when the payload looks like an automated response.
267     */
268    private function is_auto_submitted( array $payload ): bool {
269        $raw = EmailAuth::extract_mailgun_header( $payload, 'auto-submitted' );
270        if ( ! $raw ) {
271            // Generic fallback: some ESPs include it as a top-level field.
272            $raw = (string) ( $payload['Auto-Submitted'] ?? $payload['auto-submitted'] ?? '' );
273        }
274
275        $value = strtolower( trim( $raw ) );
276        return '' !== $value && 'no' !== $value;
277    }

292

295            return new WP_Error( 'agnosis_no_secret', __( 'Webhook secret not configured.', 'agnosis' ), [ 'status' => 403 ] );
