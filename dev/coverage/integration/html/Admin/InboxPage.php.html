476    private function human_date( string $datetime ): string {
477        $ts  = strtotime( $datetime );
478        $age = time() - (int) $ts;
479
480        if ( $age < 60 ) {
481            return __( 'just now', 'agnosis' );
482        }
483        if ( $age < 3600 ) {
484            $m = (int) round( $age / 60 );
485            /* translators: %d: number of minutes */
486            return sprintf( __( '%dm ago', 'agnosis' ), $m );
487        }
488        if ( $age < 86400 ) {
489            $h = (int) round( $age / 3600 );
490            /* translators: %d: number of hours */
491            return sprintf( __( '%dh ago', 'agnosis' ), $h );
492        }
493        return wp_date( 'M j', (int) $ts ) ?: $datetime;
494    }
