795    private function human_date( string $datetime ): string {
796        $ts  = strtotime( $datetime );
797        $age = time() - (int) $ts;
798
799        if ( $age < 60 ) {
800            return __( 'just now', 'agnosis' );
801        }
802        if ( $age < 3600 ) {
803            $m = (int) round( $age / 60 );
804            /* translators: %d: number of minutes */
805            return sprintf( __( '%dm ago', 'agnosis' ), $m );
806        }
807        if ( $age < 86400 ) {
808            $h = (int) round( $age / 3600 );
809            /* translators: %d: number of hours */
810            return sprintf( __( '%dh ago', 'agnosis' ), $h );
811        }
812        return wp_date( 'M j', (int) $ts ) ?: $datetime;
813    }
