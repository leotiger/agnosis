152    /**
153     * Order the agnosis_event archive chronologically by the event's own date
154     * meta (soonest first) rather than post publish date — a visitor browsing
155     * an artist's Events page wants to see what's coming up next, not what was
156     * emailed in most recently. Events with no recorded date sort last rather
157     * than being dropped from the list.
158     *
159     * Merges into whatever meta_query already exists rather than replacing it
160     * (0.9.24 fix) — this used to `$q->set('meta_query', …)` outright, which
161     * silently discarded Lingua Forge's own `_lf_lang` clause
162     * (QueryFilter::handle_pre_get_posts(), also hooked on `pre_get_posts` for
163     * the exact same is_archive() case) whenever that filter happened to run
164     * first. Agnosis's own `pre_get_posts` registration (Core\Plugin.php) is
165     * wired up after Lingua Forge's — LF is a hard dependency and loads first
166     * — so on every request this handler ran second and clobbered the
167     * language scoping outright, and every language's events piled up
168     * together on one archive page instead of just the current one's. Nesting
169     * the pre-existing meta_query as its own AND-ed group alongside the
170     * OR-relation date clause below preserves it (and anything else a future
171     * filter adds) regardless of hook registration order, rather than
172     * depending on this callback always running first.
173     */
174    public function order_events_archive( \WP_Query $q ): void {
175        if ( ! $q->is_main_query() || is_admin() || ! $q->is_post_type_archive( 'agnosis_event' ) ) {
176            return;
177        }
178
179        $q->set( 'meta_key', '_agnosis_event_date' );
180        $q->set( 'orderby', [ 'meta_value' => 'ASC', 'date' => 'DESC' ] );
181        // meta_key + orderby=>meta_value alone would silently exclude any event
182        // with no _agnosis_event_date at all — the OR below keeps them in the
183        // results (they just sort after every dated event, per 'orderby' above).
184        $date_clause = [
185            'relation' => 'OR',
186            [ 'key' => '_agnosis_event_date', 'compare' => 'EXISTS' ],
187            [ 'key' => '_agnosis_event_date', 'compare' => 'NOT EXISTS' ],
188        ];
189
190        $existing = (array) $q->get( 'meta_query' );
191        $q->set( 'meta_query', empty( $existing ) ? $date_clause : [ 'relation' => 'AND', $existing, $date_clause ] );
192    }

224        $is_artwork_loop = 'agnosis_artwork' === $post_type
225            || ( is_array( $post_type ) && in_array( 'agnosis_artwork', $post_type, true ) );
226
227        if ( ! $is_artwork_loop || ! empty( $query['author'] ) ) {

236        }
237        $query['post__not_in'] = array_merge( (array) ( $query['post__not_in'] ?? [] ), [ $current_id ] );
238
