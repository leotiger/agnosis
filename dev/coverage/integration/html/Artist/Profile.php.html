153     * Order the agnosis_event archive chronologically by the event's own date
154     * meta (soonest first) rather than post publish date — a visitor browsing
155     * an artist's Events page wants to see what's coming up next, not what was
156     * emailed in most recently. Events with no recorded date sort last rather
157     * than being dropped from the list.
158     */
159    public function order_events_archive( \WP_Query $q ): void {
160        if ( ! $q->is_main_query() || is_admin() || ! $q->is_post_type_archive( 'agnosis_event' ) ) {
161            return;
162        }
163
164        $q->set( 'meta_key', '_agnosis_event_date' );
165        $q->set( 'orderby', [ 'meta_value' => 'ASC', 'date' => 'DESC' ] );
166        // meta_key + orderby=>meta_value alone would silently exclude any event
167        // with no _agnosis_event_date at all — the OR below keeps them in the
168        // results (they just sort after every dated event, per 'orderby' above).
169        $q->set( 'meta_query', [
170            'relation' => 'OR',
171            [ 'key' => '_agnosis_event_date', 'compare' => 'EXISTS' ],
172            [ 'key' => '_agnosis_event_date', 'compare' => 'NOT EXISTS' ],
173        ] );

200    public function scope_more_works_query( array $query ): array {
201        if ( ! is_singular( 'agnosis_artwork' ) ) {
202            return $query;
203        }
204
205        $post_type       = $query['post_type'] ?? '';
206        $is_artwork_loop = 'agnosis_artwork' === $post_type
207            || ( is_array( $post_type ) && in_array( 'agnosis_artwork', $post_type, true ) );
208
209        if ( ! $is_artwork_loop || ! empty( $query['author'] ) ) {
210            return $query;
211        }
212

214        $author_id  = (int) get_post_field( 'post_author', $current_id );
215
216        if ( $author_id > 0 ) {
217            $query['author'] = $author_id;
218        }
219        $query['post__not_in'] = array_merge( (array) ( $query['post__not_in'] ?? [] ), [ $current_id ] );
220
221        return $query;
222    }

351    /**
352     * Render callback for the agnosis/artwork-title block.
353     *
354     * Outputs the artwork's canonical title (post_title, in the artist's language)
355     * as an <h1>.  When a site-language AI translation exists and differs from the
356     * original, it is rendered below as a smaller subtitle so visitors who do not
357     * speak the artist's language can still understand the work's name.
358     *
359     * Example output for a Chinese artwork on an English site:
360     *
361     *   <hgroup class="agnosis-artwork-title">
362     *     <h1 class="agnosis-artwork-title__original">秘密花園</h1>
363     *     <p  class="agnosis-artwork-title__translation">The Secret Garden</p>
364     *   </hgroup>
365     *
366     * @param array<string, mixed> $attrs   Block attributes (unused).
367     * @param string               $content Inner block content (unused).
368     * @param \WP_Block            $block   Block instance (provides postId context).
369     * @return string HTML output.
370     */
371    public function render_artwork_title( array $attrs, string $content, \WP_Block $block ): string {
372        $post_id = (int) ( $block->context['postId'] ?? get_the_ID() );
373        $post    = get_post( $post_id );
374
375        if ( ! $post ) {
376            return '';
377        }
378
379        $original    = trim( $post->post_title );
380        $translation = trim( (string) get_post_meta( $post_id, '_agnosis_translated_title', true ) );
381
382        // Dual-title editing (Phase 3, audit §7c/§7d): post_title is the artist's
383        // own words — the only part of this block that's ever directly editable.
384        // _agnosis_translated_title is a separate AI-generated value the artist
385        // doesn't type into here; it's regenerated automatically (see
386        // ContentEditor::propagate_title()) whenever the original title changes.
387        $h1 = ContentEditor::is_editable_by_current_user( $post_id )
388            ? sprintf(
389                '<h1 class="agnosis-artwork-title__original agnosis-editable" data-agnosis-edit-field="title" data-agnosis-post-id="%d" style="font-style:italic;font-weight:300;%s">%s</h1>',
390                $post_id,
391                '' === $translation || $translation === $original ? '' : 'margin:0 0 0.25em;',
392                esc_html( $original )
393            )
394            : sprintf(
395                '<h1 class="agnosis-artwork-title__original" style="font-style:italic;font-weight:300;%s">%s</h1>',
396                '' === $translation || $translation === $original ? '' : 'margin:0 0 0.25em;',
397                esc_html( $original )
398            );
399
400        // When both strings are identical (same language, no translation stored, or
401        // translation not yet run), render a plain heading — no extra markup.
402        if ( '' === $translation || $translation === $original ) {
403            return $h1;
404        }
405
406        // Colour: NOT --wp--preset--color--secondary — that token is a ~6%-opacity
407        // background wash (theme.json), not a text colour, and rendered this
408        // translated title all but invisible against the dark artwork-page
409        // header (reported 2026-07-07). The theme's own established pattern for
410        // legible-but-muted text over that same dark header (post-date block,
411        // breadcrumb, pagination — see agnosis-theme's single-agnosis_artwork.html
412        // and style.css) is full-strength --foreground at reduced opacity, not a
413        // separate low-alpha colour token. Using a higher opacity (0.75) than
414        // those nav-ish elements (0.4-0.5) since this is actual translated
415        // content — the work's official site-language title — not incidental
416        // chrome, and should be comfortably readable, not merely present.
417        // Font-size bumped one step (small -> medium) per the same report.
418        return sprintf(
419            '<hgroup class="agnosis-artwork-title" style="margin:0;">%s'
420            . '<p class="agnosis-artwork-title__translation" style="margin:0.15em 0 0;font-size:var(--wp--preset--font-size--medium);color:var(--wp--preset--color--foreground);opacity:0.75;font-style:normal;">%s</p>'
421            . '</hgroup>',
422            $h1,
423            esc_html( $translation )
424        );
425    }
