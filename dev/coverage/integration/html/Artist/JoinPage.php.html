272    public static function resolve_success_url( string $lang = '' ): string {
273        $value = get_option( 'agnosis_join_success_url', '' );
274
275        if ( ! is_numeric( $value ) ) {
276            // Back-compat: pre-existing raw URL string from before this setting
277            // became a page selector — not a page ID, nothing to translate.
278            return (string) $value;
279        }
280
281        $page_id = (int) $value;
282        if ( $page_id <= 0 ) {
283            return '';
284        }
285
286        if ( '' !== $lang && LinguaForge::is_active() && function_exists( 'linguaforge_get_translations' ) ) {
287            // phpcs:ignore WordPress.NamingConventions.PrefixAllGlobals.NonPrefixedFunctionFound -- calling Lingua Forge's public API; prefix belongs to that plugin.
288            $translations = linguaforge_get_translations( $page_id );
289            if ( isset( $translations[ $lang ] ) && 'publish' === get_post_status( (int) $translations[ $lang ] ) ) {
290                $translated_permalink = get_permalink( (int) $translations[ $lang ] );
291                if ( $translated_permalink ) {
292                    return $translated_permalink;
293                }
294            }
295        }
296
297        $permalink = get_permalink( $page_id );
298        return $permalink ?: '';
