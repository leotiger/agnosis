307    /**
308     * Flatten a full exception chain into a single readable string.
309     *
310     * webklex wraps low-level stream/SSL errors inside ConnectionFailedException,
311     * so getMessage() alone returns "connection failed" without the actual cause.
312     * Walking getPrevious() surfaces the real OpenSSL / stream error message.
313     */
314    private static function exception_chain( \Throwable $e ): string {
315        $parts = [];
316        $seen  = [];
317        while ( null !== $e ) {
318            $msg = trim( $e->getMessage() );
319            // Deduplicate adjacent identical messages.
320            if ( '' !== $msg && ! in_array( $msg, $seen, true ) ) {
321                $parts[] = $msg;
322                $seen[]  = $msg;
323            }
324            $e = $e->getPrevious();
325        }
326        return implode( ' → ', $parts );
327    }

355        $ssl     = (bool) get_option( 'agnosis_imap_ssl', true );
356        $no_cert = (bool) get_option( 'agnosis_imap_novalidate_cert', false );
357        $user    = (string) get_option( 'agnosis_imap_user' );
358        $pass    = (string) get_option( 'agnosis_imap_pass' );
359
360        // ClientManager is required — instantiating Client directly passes null
361        // as the default_config argument, triggering a TypeError in v5.
362        $manager = new ClientManager( [] );
363
364        $client = $manager->make( [
365            'host'          => $host,
366            'port'          => $port,
367            // 'imap' uses the native PHP stream driver (SNI-capable).
368            // Never use 'legacy-imap' — that falls back to imap_open() which has no SNI.
369            'protocol'      => 'imap',
370            'encryption'    => $ssl ? 'ssl' : false,
371            'validate_cert' => ! $no_cert,
372            'username'      => $user,
373            'password'      => $pass,
374        ] );
375
376        $client->connect();
377        return $client;
378    }
379
380    /**
381     * Fetch recent messages and enqueue any not yet in the pipeline.
382     *
383     * We query ALL messages within the retention window rather than only UNSEEN
384     * ones. This avoids a race condition where a concurrent mail client (e.g.
385     * Apple Mail) re-marks messages as \Seen between our clearFlag and our query,
386     * causing the UNSEEN search to return 0.  The database `is_already_queued()`
387     * state machine is the authoritative idempotency guard — IMAP flags are not.
388     *
389     * @param Folder $folder
390     */
391    private function process_messages( Folder $folder ): void {
392        // ---- Build the query — headers-first, UID-sequenced -----------------
393        //
394        // fetchBody(false): webklex fetches only RFC-822 headers on the first
395        // IMAP FETCH.  The body is pulled lazily when parse_imap_message() is
396        // called below — and only for messages that clear all cheap gates.
397        //
398        // setSequence(ST_UID): the IMAP server numbers messages two ways:
399        //   • Message Sequence Numbers (MSN) — renumber when older mail is deleted.
400        //   • UIDs — stable for the lifetime of a mailbox.
401        // Operating in UID mode means our cursor survives expunges and reconnects.
402        //
403        // UID cursor: on every run after the first we send `UID N+1:*` which tells
404        // the server to return only messages whose UID is strictly greater than the
405        // last one we processed.  First run (or cursor reset) falls back to the
406        // configured retention window so we do not skip legitimate back-dated mail.
407
408        $last_uid = max( 0, (int) get_option( 'agnosis_imap_last_uid', 0 ) );
409        $messages = $this->query_messages( $folder, $last_uid );
410        $max_uid_seen = $last_uid;
411
412        $goodbye_addr = strtolower( trim( (string) get_option( 'agnosis_email_goodbye', '' ) ) );
413
414        foreach ( $messages as $message ) {
415            try {
416                $uid = (string) $message->getUid();
417
418                // Track the highest UID we have seen in this poll so we can
419                // advance the cursor even for messages we decide to skip.
420                $numeric_uid = (int) $uid;
421                if ( $numeric_uid > $max_uid_seen ) {
422                    $max_uid_seen = $numeric_uid;
423                }
424
425                if ( $this->is_already_queued( $uid ) ) {
426                    continue; // DB state machine handles all cases.
427                }
428
429                // ---- Cheap header checks (before body download) --------------------
430                // Read From: and To: from headers — no body fetch needed. These are
431                // fast IMAP FETCH (RFC 822 header) operations. All expensive work
432                // (body download, attachment parsing, AI) only runs for admitted senders
433                // that also pass throttle and auth checks.
434
435                $from_list  = $message->getFrom()->toArray();
436                $from_email = $from_list ? sanitize_email( (string) $from_list[0]->mail ) : '';
437
438                // --- Goodbye alias: self-removal request (no attachment required) ---
439                if ( $goodbye_addr ) {
440                    $to_list = $message->getTo()->toArray();
441                    $msg_to  = $to_list ? strtolower( sanitize_email( (string) $to_list[0]->mail ) ) : '';
442                    if ( $msg_to === $goodbye_addr ) {
443                        $this->handle_goodbye_email( $from_email, $uid );
444                        continue;
445                    }
446                }
447
448                // --- Cheap gate 1: admitted sender (header-only) -------------------
449                $artist_user = $from_email ? get_user_by( 'email', $from_email ) : null;
450                $artist_id   = $artist_user ? $artist_user->ID : null;
451
452                if ( null === $artist_id ) {
453                    Logger::warning( 'Skipped UID ' . $uid . ': unregistered sender <' . $from_email . '>.', 'inbox' );
454                    $this->mark_no_artwork( $uid, null, 'unregistered_sender' );
455                    continue;
456                }
457
458                if ( ! $this->is_admitted_artist( $artist_id ) ) {
459                    Logger::warning( 'Skipped UID ' . $uid . ': sender <' . $from_email . '> (user #' . $artist_id . ') is not admitted.', 'inbox' );
460                    $this->mark_no_artwork( $uid, $artist_id, 'not_admitted' );
461                    continue;
462                }
463
464                // --- Cheap gate 2: per-sender intake throttle ----------------------
465                $sender_limit  = max( 1, (int) get_option( 'agnosis_intake_per_sender_limit', 5 ) );
466                $throttle      = RateLimiter::check_sender( 'email_intake', $from_email, $sender_limit, HOUR_IN_SECONDS );
467                if ( is_wp_error( $throttle ) ) {
468                    Logger::warning( 'Skipped UID ' . $uid . ': sender <' . $from_email . '> throttled (' . $sender_limit . '/hour limit).', 'inbox' );
469                    $this->mark_no_artwork( $uid, $artist_id, 'throttled' );
470                    continue;
471                }
472
473                // --- Cheap gate 3: SPF/DKIM (opt-in) -------------------------------
474                if ( get_option( 'agnosis_require_email_auth' ) ) {
475                    $auth_raw = '';
476                    try {
477                        $auth_raw = (string) $message->getHeader()->get( 'authentication-results' );
478                    } catch ( \Throwable $e ) { // phpcs:ignore Generic.CodeAnalysis.EmptyStatement.DetectedCatch -- header absent or library threw; $auth_raw stays ''
479                    }
480                    if ( $auth_raw ) {
481                        $verdicts = EmailAuth::check_header( $auth_raw );
482                        if ( ! EmailAuth::passes( $verdicts ) ) {
483                            Logger::warning(
484                                sprintf( 'Skipped UID ' . $uid . ': <' . $from_email . '> failed auth — SPF=%s DKIM=%s.', $verdicts['spf'] ?: 'none', $verdicts['dkim'] ?: 'none' ),
485                                'inbox'
486                            );
487                            $this->mark_no_artwork( $uid, $artist_id, 'auth_failed' );
488                            continue;
489                        }
490                    } else {
491                        Logger::warning( 'Skipped UID ' . $uid . ': auth required but no Authentication-Results header.', 'inbox' );
492                        $this->mark_no_artwork( $uid, $artist_id, 'auth_failed' );
493                        continue;
494                    }
495                }
496
497                // ---- Full parse (body + attachments) — admitted senders only ------
498                $submission = $this->parser->parse_imap_message( $message );
499
500                if ( null === $submission ) {
501                    // No valid image attachments — mark it so we don't re-check it. The
502                    // sender IS a known, admitted artist here (both gates above already
503                    // passed) — persist $artist_id so the Inbox admin table shows who
504                    // this was instead of "unregistered sender", and so a later
505                    // auto-retry (is_already_queued()'s 'failed' branch) doesn't have
506                    // PostCreator wrongly re-reject a real artist as unadmitted.
507                    Logger::info( 'Skipped UID ' . $uid . ': no valid image attachments.', 'inbox' );
508
509                    if ( Debug::enabled() ) {
510                        // Parser::parse_imap_message() already wrote a detailed
511                        // 'parser-attachments-*' dump (raw MIME structure, per-
512                        // attachment MIME sniff/declared/disposition/size, and
513                        // why each one was accepted or rejected) for this exact
514                        // UID just above, in the call this branch is reacting
515                        // to. This companion file just makes that dump easy to
516                        // find from a directory listing sorted by time — the
517                        // two should have matching timestamps.
518                        Debug::write(
519                            'inbox-skip-no-attachments',
520                            sprintf(
521                                "UID: %s\nFrom: %s\nSubject: %s\nArtist user ID: %d\n\nSee the parser-attachments-* dump written immediately before this one (same poll cycle) for the full MIME/attachment trace.",
522                                $uid,
523                                $from_email,
524                                (string) $message->getSubject(),
525                                $artist_id
526                            )
527                        );
528                    }
529
530                    $this->mark_no_artwork( $uid, $artist_id, 'no_attachments' );
531
532                    /**
533                     * Fires when an admitted artist's email was received but contained no
534                     * recognizable image/audio/video attachment (e.g. a photo inserted
535                     * inline rather than properly attached).
536                     *
537                     * @param int    $artist_id WordPress user ID of the sender.
538                     * @param string $uid       IMAP message UID (for logging/reference only).
539                     */
540                    do_action( 'agnosis_submission_no_attachment', $artist_id, $uid );
541                    continue;
542                }
543
544                $this->enqueue( $uid, $submission );
545
546            } catch ( \Throwable $e ) {
547                Logger::error( 'Error processing message UID ' . ( $uid ?? '?' ) . ': ' . $e->getMessage(), 'inbox' );
548            }
549        }
550
551        // Advance the UID cursor so the next poll only fetches genuinely new mail.
552        if ( $max_uid_seen > $last_uid ) {
553            update_option( 'agnosis_imap_last_uid', $max_uid_seen );
554            Logger::info( 'Poll: UID cursor advanced to ' . $max_uid_seen . '.', 'inbox' );
555        }
556    }
557
558    /**
559     * Build the IMAP message query for this poll.
560     *
561     * Extracted as a protected method so tests can subclass Inbox and inject a
562     * controlled message collection without requiring a live IMAP connection.
563     *
564     * When $last_uid > 0 we send `UID N+1:*` to the server, retrieving only
565     * messages whose UID is strictly greater than the last processed one.
566     * On the first run ($last_uid === 0) we fall back to a date-bounded window
567     * so we do not skip legitimate back-dated mail.
568     *
569     * @param Folder $folder   Webklex folder to query.
570     * @param int    $last_uid Last UID persisted from the previous poll (0 = first run).
571     * @return \Webklex\PHPIMAP\Support\MessageCollection
572     */
573    protected function query_messages( Folder $folder, int $last_uid ): \Webklex\PHPIMAP\Support\MessageCollection {
574        $base = $folder->query(); // WhereQuery — do not chain through Query-typed fluent setters.
575        $base->setSequence( IMAP::ST_UID );
576        $base->fetchBody( false );
577
578        if ( $last_uid > 0 ) {
579            $messages = $base->whereUid( ( $last_uid + 1 ) . ':*' )->get();
580            Logger::info(
581                sprintf( 'Poll: UID cursor at %d — fetched %d new message(s).', $last_uid, $messages->count() ),
582                'inbox'
583            );
584            return $messages;
585        }
586
587        $days  = max( 1, (int) get_option( 'agnosis_imap_cleanup_days', 30 ) );
588        $since = new \DateTime( '-' . $days . ' days' );
589        $messages = $base->since( $since )->get();
590        Logger::info(
591            sprintf( 'Poll: no UID cursor — scanning %d message(s) from last %d day(s).', $messages->count(), $days ),
592            'inbox'
593        );
594        return $messages;
595    }
596
597    /**
598     * Handle a goodbye email: trigger the self-removal confirmation flow for
599     * the sending artist, then mark the message so it won't be re-processed.
600     *
601     * @param string $from_email Sender email address.
602     * @param string $uid        IMAP message UID (used for deduplication).
603     */
604    private function handle_goodbye_email( string $from_email, string $uid ): void {
605        $user = get_user_by( 'email', $from_email );
606
607        if ( ! $user || ! $this->is_admitted_artist( $user->ID ) ) {
608            Logger::warning(
609                'Goodbye email from non-artist <' . $from_email . '> — ignored.',
610                'inbox'
611            );
612            $this->mark_no_artwork( $uid, $user ? (int) $user->ID : null, 'goodbye_non_artist' );
613            return;
614        }
615
616        $departure = new Departure();
617        $ok        = $departure->initiate_removal_for_user( $user->ID );
618
619        if ( $ok ) {
620            Logger::info(
621                'Goodbye email from <' . $from_email . '> (user #' . $user->ID . '): confirmation sent.',
622                'inbox'
623            );
624        } else {
625            Logger::warning(
626                'Goodbye email from <' . $from_email . '> (user #' . $user->ID . '): no active membership — ignored.',
627                'inbox'
628            );
629        }
630
631        $this->mark_no_artwork( $uid, $user->ID, 'goodbye_handled' );
632    }
633
634    /**
635     * Human-readable error text per skip reason — shown in the Inbox admin
636     * table's Error column, and (via $reason) drives whether the sender's
637     * identity is worth keeping on the row.
638     *
639     * Every reason gets its own accurate string. Previously every call site
640     * shared one generic "no valid artwork or unregistered sender" message,
641     * which made a content problem (no attachment) indistinguishable from an
642     * identity problem (unknown sender) — see mark_no_artwork() below.
643     *
644     * @var array<string, string>
645     */
646    private const SKIP_REASONS = [
647        'unregistered_sender'  => 'Skipped: sender is not a registered WordPress user.',
648        'not_admitted'         => 'Skipped: sender is registered but not an admitted artist.',
649        'throttled'            => 'Skipped: sender exceeded the per-hour intake limit.',
650        'auth_failed'          => 'Skipped: message failed SPF/DKIM authentication.',
651        'no_attachments'       => 'Skipped: no valid image, audio, or video attachment found in the message.',
652        'goodbye_non_artist'   => 'Skipped: goodbye request from a non-artist sender.',
653        'goodbye_handled'      => 'Goodbye request processed — self-removal confirmation sent.',
654    ];
655
656    /**
657     * Record a skipped message in the queue table so we never re-check it.
658     *
659     * Uses status 'failed' with a reason-specific error so it appears in the
660     * queue and can be inspected, but won't be picked up by PostCreator.
661     * INSERT IGNORE is used because a row may already exist (e.g. previous skip).
662     *
663     * $artist_id is persisted whenever it's already known (i.e. the sender did
664     * resolve to a WP user, even if not an admitted artist) purely so the Inbox
665     * admin table can show who actually sent it instead of always falling back
666     * to "unregistered sender". These rows are stored with raw_email = '{}' —
667     * there was never a real submission to preserve — and is_already_queued()
668     * deliberately does NOT auto-retry them (unlike a genuine PostCreator
669     * processing failure, which keeps the full original submission and is
670     * worth retrying): the underlying email never changes, so a retry can't
671     * produce a different outcome, and resetting one of these to 'pending'
672     * previously handed PostCreator::handle() a genuinely empty submission —
673     * which slipped past every "no attachment" guard and published a blank,
674     * untitled draft. If a sender is later admitted, they simply need to
675     * resend their email — see is_already_queued()'s 'failed' case.
676     *
677     * @param string   $uid       IMAP message UID.
678     * @param int|null $artist_id WP user ID of the sender, if resolved. Null when
679     *                            the sender never matched a WP account at all.
680     * @param string   $reason    Key into self::SKIP_REASONS.
681     */
682    private function mark_no_artwork( string $uid, ?int $artist_id = null, string $reason = 'unregistered_sender' ): void {
683        global $wpdb;
684
685        $error = self::SKIP_REASONS[ $reason ] ?? self::SKIP_REASONS['unregistered_sender'];
686
687        // phpcs:ignore WordPress.DB.DirectDatabaseQuery.DirectQuery, WordPress.DB.DirectDatabaseQuery.NoCaching -- custom table insert; INSERT IGNORE is idempotent.
688        $wpdb->query(
689            $wpdb->prepare(
690                "INSERT IGNORE INTO {$wpdb->prefix}agnosis_queue
691                 (message_uid, artist_id, status, raw_email, error)
692                 VALUES (%s, %s, 'failed', '{}', %s)",
693                $uid,
694                null !== $artist_id ? (string) $artist_id : null,
695                $error
696            )
697        );
698    }
699
700    /**
701     * Check whether a WP user is an admitted Agnosis artist (or an admin).
702     *
703     * Delegates to the shared Admission::is_admitted_artist() so the logic
704     * stays in one place across all intake paths.
705     */
706    private function is_admitted_artist( int $user_id ): bool {
707        return Admission::is_admitted_artist( $user_id );
708    }
709
710    /**
711     * Check whether a message UID is already handled — and auto-heal broken states.
712     *
713     * Returns true if the message should be skipped by process_messages(); false if

715     * Process Pending Queue can pick them up without needing a new IMAP fetch.

724     *  - failed         → reset to pending, true  (retry via Process Queue)

726     * Returning true in all non-null cases means process_messages() marks the IMAP
727     * message as \Seen and moves on — the queued raw_email data is sufficient to

785                );

788                return true; // Reset — pipeline will re-run.
789
790            case 'failed':
791                // Two very different things share the 'failed' status:
792                //   - A real PostCreator::handle() processing failure (AI timeout,
793                //     provider error, etc.) — mark() only ever touches status/error/
794                //     post_id, so raw_email still holds the full original submission.
795                //     Retrying this is exactly the point: the same content might
796                //     succeed on a second attempt.
797                //   - An intake-gate skip recorded by mark_no_artwork() (unregistered

817                wp_schedule_single_event( time(), 'agnosis_publish_submission', [ $queue_id ] );
818                return true; // Reset — pipeline will retry.
819
820            default:
821                return true;
822        }
