371    /**
372     * Handle a bounce/complaint event webhook POST.
373     *
374     * Recognizes three payload shapes (each ESP's own native format — this
375     * plugin does not ask the operator to reshape anything):
376     *   - Postmark: a single JSON object, `RecordType` is `'Bounce'` or `'SpamComplaint'`.
377     *   - Mailgun: `event`/`recipient`/`severity`, either at the top level
378     *     (classic form-encoded webhooks) or nested under `event-data`
379     *     (current JSON webhooks) — both shapes are checked.
380     *   - SendGrid: a raw JSON array of event objects, one per address.
381     *
382     * Only HARD bounces and spam complaints suppress anything — a soft/
383     * transient bounce (mailbox full, temporary server issue) is not "this
384     * address is dead" and must not stop future sends on one blip; each
385     * provider's own hard/soft distinction is checked before calling
386     * BounceHandler::record().
387     *
388     * Always returns 200 (never a queue-worthy retry signal) — an
389     * unrecognized payload shape or an address matching neither a
390     * subscriber nor an artist is not an error the ESP should retry.
391     */
392    public function handle_bounce_event( WP_REST_Request $request ): WP_REST_Response {
393        $json_params = $request->get_json_params();
394
395        // SendGrid posts a raw JSON array — not an object — so it never
396        // resembles the other two shapes below, which are always keyed arrays.
397        if ( is_array( $json_params ) && array_is_list( $json_params ) ) {
398            $processed = 0;
399            foreach ( $json_params as $event ) {
400                if ( is_array( $event ) && $this->process_sendgrid_event( $event ) ) {
401                    ++$processed;
402                }
403            }
404            return new WP_REST_Response( [ 'status' => 'processed', 'count' => $processed ], 200 );
405        }
406
407        $payload = $request->get_params();
408
409        if ( isset( $payload['RecordType'] ) ) {
410            $this->process_postmark_event( $payload );
411            return new WP_REST_Response( [ 'status' => 'processed' ], 200 );
412        }
413
414        // Mailgun: prefer the nested 'event-data' object (current JSON
415        // webhooks) when present, otherwise treat the top level as the event
416        // itself (classic form-encoded webhooks — the same shape handle()'s
417        // own Mailgun signature verification above already expects).
418        $event_data = is_array( $payload['event-data'] ?? null ) ? $payload['event-data'] : $payload;
419        if ( isset( $event_data['event'] ) ) {
420            $this->process_mailgun_event( $event_data );
421            return new WP_REST_Response( [ 'status' => 'processed' ], 200 );
422        }
423
424        Logger::warning( 'Webhook: /email/events received a payload that matched no known ESP event shape.', 'webhook' );
425        return new WP_REST_Response( [ 'status' => 'ignored', 'reason' => 'unrecognized_payload' ], 200 );
426    }

428    /**
429     * @param array<string, mixed> $payload Postmark's own single-event JSON object.
430     */
431    private function process_postmark_event( array $payload ): void {
432        $record_type = (string) ( $payload['RecordType'] ?? '' );
433        $email       = sanitize_email( (string) ( $payload['Email'] ?? '' ) );
434
435        if ( '' === $email ) {
436            return;
437        }
438
439        if ( 'SpamComplaint' === $record_type ) {
440            BounceHandler::record( $email, 'complaint', 'webhook' );
441            return;
442        }
443
444        if ( 'Bounce' === $record_type ) {
445            // Postmark's 'Type' field distinguishes 'HardBounce' from soft/
446            // transient variants (e.g. 'SoftBounce', 'Transient') — only the
447            // former means the address is actually dead.
448            $bounce_type = (string) ( $payload['Type'] ?? '' );
449            if ( false !== stripos( $bounce_type, 'hard' ) ) {
450                BounceHandler::record( $email, 'bounce', 'webhook' );
451            }
452        }
453    }
454
455    /**
456     * @param array<string, mixed> $event_data Mailgun's event object — either the
457     *                                          top-level payload (classic) or its
458     *                                          nested 'event-data' (current JSON).
459     */
460    private function process_mailgun_event( array $event_data ): void {
461        $event     = strtolower( (string) ( $event_data['event'] ?? '' ) );
462        $recipient = sanitize_email( (string) ( $event_data['recipient'] ?? '' ) );
463
464        if ( '' === $recipient ) {
465            return;
466        }
467
468        if ( 'complained' === $event ) {
469            BounceHandler::record( $recipient, 'complaint', 'webhook' );
470            return;
471        }
472
473        if ( 'failed' === $event ) {
474            // 'permanent' severity = hard bounce; 'temporary' is a delivery
475            // delay/retry Mailgun itself is still working on, not a dead address.
476            $severity = strtolower( (string) ( $event_data['severity'] ?? '' ) );
477            if ( 'permanent' === $severity ) {
478                BounceHandler::record( $recipient, 'bounce', 'webhook' );
479            }
480        }
481    }
482
483    /**
484     * @param array<string, mixed> $event One element of SendGrid's raw JSON event array.
485     * @return bool True when this event was recognized and acted on.
486     */
487    private function process_sendgrid_event( array $event ): bool {
488        $type  = strtolower( (string) ( $event['event'] ?? '' ) );
489        $email = sanitize_email( (string) ( $event['email'] ?? '' ) );
490
491        if ( '' === $email ) {
492            return false;
493        }
494
495        if ( 'spamreport' === $type ) {
496            BounceHandler::record( $email, 'complaint', 'webhook' );
497            return true;
498        }
499
500        if ( 'bounce' === $type ) {
501            // SendGrid's own 'type' field distinguishes a hard 'bounce' from a
502            // transient 'blocked' — SendGrid itself retries the latter, so
503            // only 'bounce' means the address is actually dead.
504            $sg_type = strtolower( (string) ( $event['type'] ?? '' ) );
505            if ( 'bounce' === $sg_type ) {
506                BounceHandler::record( $email, 'bounce', 'webhook' );
507                return true;
508            }
509        }
510
511        return false;
512    }

514    /**
515     * Opt-in SPF/DKIM authentication gate (fourth audit §3b).
516     *
517     * Evaluated for EVERY request in handle() — including the goodbye@/
518     * community@ aliases — before any routing decision is made, not just
519     * requests that fall through to the normal artwork/bio/event pipeline.
520     * Returns true immediately (no-op) when the `agnosis_require_email_auth`
521     * option is off, which it is by default.
522     *
523     * @param array<string, mixed> $payload    Raw webhook POST params.
524     * @param string               $from_email Sender address, used only for logging.
525     * @return bool True when the request passes (or auth is not required at all).
526     */
527    private function passes_email_auth( array $payload, string $from_email ): bool {
528        if ( ! get_option( 'agnosis_require_email_auth' ) ) {
529            return true;
530        }
531
532        // When enabled, reject messages that fail both SPF and DKIM. This stops
533        // spoofed From: addresses — an attacker who knows an artist's email but
534        // does not control their domain will fail both checks.
535        $auth_header = EmailAuth::extract_from_mailgun_payload( $payload );
536        // Generic fallback: some ESPs include it as a top-level field.
537        if ( ! $auth_header ) {
538            $auth_header = (string) ( $payload['Authentication-Results'] ?? $payload['authentication-results'] ?? '' );
539        }
540
541        if ( ! $auth_header ) {
542            // No Authentication-Results header found — cannot verify. When auth
543            // is required but not available, reject to be safe.
544            Logger::warning( 'Webhook: rejected <' . $from_email . '> — authentication required but no Authentication-Results header found.', 'webhook' );
545            return false;
546        }
547
548        $verdicts = EmailAuth::check_header( $auth_header );
549        if ( ! EmailAuth::passes( $verdicts ) ) {
550            Logger::warning(
551                sprintf(
552                    'Webhook: rejected <' . $from_email . '> — SPF=%s DKIM=%s DMARC=%s.',
553                    $verdicts['spf'] ?: 'none',
554                    $verdicts['dkim'] ?: 'none',
555                    $verdicts['dmarc'] ?: 'none'
556                ),
557                'webhook'
558            );
559            return false;
560        }
561
562        return true;
563    }

565    /**
566     * Whether a webhook payload carries an `Auto-Submitted` header indicating
567     * it is an automated response (vacation auto-reply, mailing-list bounce,
568     * etc.) rather than a human-written email (fourth audit §3c).
569     *
570     * Per RFC 3834, genuine auto-responses set this header to a value other
571     * than `no` (typically `auto-replied`) — normal mail either omits the
572     * header or, less commonly, sets it explicitly to `no`; only a present,
573     * non-`no` value trips this guard, so an absent header (the overwhelming
574     * majority of real mail) is not treated as suspicious.
575     *
576     * @param array<string, mixed> $payload Raw webhook POST params.
577     * @return bool True when the payload looks like an automated response.
578     */
579    private function is_auto_submitted( array $payload ): bool {
580        $raw = EmailAuth::extract_mailgun_header( $payload, 'auto-submitted' );
581        if ( ! $raw ) {
582            // Generic fallback: some ESPs include it as a top-level field.
583            $raw = (string) ( $payload['Auto-Submitted'] ?? $payload['auto-submitted'] ?? '' );
584        }
585
586        $value = strtolower( trim( $raw ) );
587        return '' !== $value && 'no' !== $value;
588    }

613    private const ALIAS_STATUSES = IntakeGates::SHARED_ALIAS_STATUSES;

615    /**
616     * Record a goodbye@/community@ webhook event in the queue table so it's
617     * visible in the Inbox admin table (fifth audit §5a) — previously these
618     * events existed only as a REST response and, at best, a Logs-tab entry,
619     * so an operator debugging "my goodbye email did nothing" got a different
620     * (worse) diagnostic experience on webhook sites than on IMAP sites, where
621     * Inbox::mark_no_artwork() records every such outcome as a queue row.
622     *
623     * Deliberately scoped to only the goodbye@/community@ alias reasons, not
624     * the full submission pipeline's own skip reasons (not_admitted,
625     * throttled, auth_failed, no_attachments) — those return a REST response
626     * from handle() well before a $submission exists, and are not what this
627     * finding was about.
628     *
629     * INSERT IGNORE (mirrors mark_no_artwork()) — a retry of the same signed
630     * webhook request already dedupes on message_uid, so this never produces
631     * a duplicate row for the same physical email.
632     *

710    /**
711     * Verify HMAC signature sent by the webhook provider.
712     *
713     * Each provider has its own signing scheme; we support Mailgun and a generic
714     * X-Agnosis-Signature header (HMAC-SHA256 of the raw body).
715     *
716     * Replay protection (fifth audit §3a): both schemes below now also check
717     * that the signed timestamp is fresh (TIMESTAMP_FRESHNESS_SECONDS either
718     * direction) and remember the specific signed request for
719     * REPLAY_MEMORY_SECONDS so it can't be replayed a second time inside that
720     * freshness window. Previously a valid signed triple/pair validated
721     * forever — anyone who ever captured one (proxy logs, a misconfigured
722     * CDN, the artist's own outbox if they run the ESP) could replay it
723     * indefinitely, each replay re-spending the AI pipeline (see handle()'s
724     * queue UID fix, same audit finding, for the other half of this).
725     *
726     * Rate-limit ordering (sixth audit §3c): the signature is now matched
727     * (via match_signature() below) BEFORE any rate limit is applied, and the
728     * matched/unmatched outcomes are checked against two separate buckets —
729     * see UNVERIFIED_RATE_LIMIT/VERIFIED_RATE_LIMIT above. Previously a single
730     * shared bucket was checked first, keyed by IP; on a host where a
731     * misconfigured reverse proxy collapses every REMOTE_ADDR into one value,
732     * that made the bucket effectively global, so an unauthenticated flood of
733     * garbage could starve the ESP's own legitimate signed traffic in the
734     * same window.
735     */
736    public function verify_signature( WP_REST_Request $request ): bool|WP_Error {
737        $secret = get_option( 'agnosis_webhook_secret', '' );
738
739        if ( empty( $secret ) ) {
740            // No secret configured → reject all webhook requests. Not part of
741            // the flood-vs-legitimate-traffic bucket split below — a missing
742            // secret is an operator misconfiguration, not the abuse case §3c
743            // is about, so it isn't worth spending a rate-limit slot on.
744            return new WP_Error( 'agnosis_no_secret', __( 'Webhook secret not configured.', 'agnosis' ), [ 'status' => 403 ] );

746
747        $match = $this->match_signature( $request, $secret );

749        // Bucket split (audit §3c) — see the class docblock on
750        // UNVERIFIED_RATE_LIMIT/VERIFIED_RATE_LIMIT above for why.
751        $rate = $match['matched']
752            ? RateLimiter::check( 'email_inbound_verified', self::VERIFIED_RATE_LIMIT, self::RATE_LIMIT_WINDOW )
753            : RateLimiter::check( 'email_inbound_unverified', self::UNVERIFIED_RATE_LIMIT, self::RATE_LIMIT_WINDOW );

778    /**
779     * Cryptographically match a request's signature against both supported
780     * schemes, without any rate-limiting or replay/freshness side effects —
781     * those happen in verify_signature() above, once, using this result.
782     *
783     * @return array{attempted: bool, matched: bool, timestamp: string, replay_key: string}
784     */
785    private function match_signature( WP_REST_Request $request, string $secret ): array {
786        // --- Generic HMAC-SHA256 (default / custom senders) ---
787        $signature = $request->get_header( 'X-Agnosis-Signature' );
788        if ( $signature ) {
789            $body = $request->get_body();
790
791            // New, replay-protected form: an explicit X-Agnosis-Timestamp header
792            // folded into the HMAC input, so it can't be stripped by a replay

794            // senders that don't send it yet — those fall through to the
795            // original body-only scheme below, unchanged, still inherently
796            // replayable (no timestamp element exists to check without it).
797            $timestamp_header = $request->get_header( 'X-Agnosis-Timestamp' );
798            if ( $timestamp_header ) {

800                return [

802                    'matched'    => hash_equals( $expected, $signature ),
803                    'timestamp'  => $timestamp_header,
804                    'replay_key' => 'generic|' . $signature,
805                ];
806            }

808            $expected = hash_hmac( 'sha256', $body, $secret );

811                'matched'    => hash_equals( $expected, $signature ),
812                'timestamp'  => '',
813                'replay_key' => '',

825                'attempted'  => true,
826                'matched'    => hash_equals( $expected, $mg_signature ),
827                'timestamp'  => $mg_timestamp,

832        return [ 'attempted' => false, 'matched' => false, 'timestamp' => '', 'replay_key' => '' ];
833    }

835    /**
836     * Shared timestamp-freshness + replay-memory check for both signing
837     * schemes above (fifth audit §3a). Called only AFTER the HMAC signature
838     * itself has already been verified — a request that fails this check was
839     * genuinely signed by someone holding the secret at some point, just not
840     * recently (or already used) enough to trust now.
841     *
842     * @param string $timestamp_raw  The signed timestamp, as supplied by the sender.
843     * @param string $replay_key_raw A value unique to this specific signed request
844     *                               (e.g. Mailgun's one-time `token`, or the
845     *                               signature itself for the generic scheme) —
846     *                               remembered so the exact same request can't
847     *                               be replayed twice inside the freshness window.
848     * @return true|WP_Error
849     */
850    private function check_replay_freshness( string $timestamp_raw, string $replay_key_raw ): bool|WP_Error {
851        if ( ! ctype_digit( $timestamp_raw ) || abs( time() - (int) $timestamp_raw ) > self::TIMESTAMP_FRESHNESS_SECONDS ) {
852            return new WP_Error(
853            'agnosis_webhook_stale_timestamp',
854            __( 'Webhook request timestamp is missing, malformed, or too old.', 'agnosis' ),
855            [ 'status' => 403 ]
856            );
857        }
858
859        $replay_key = 'agnosis_wh_' . md5( $replay_key_raw );
860        if ( get_transient( $replay_key ) ) {
861            Logger::warning( 'Webhook: rejected a replayed request (same signed timestamp/token seen before).', 'webhook' );
862            return new WP_Error(
863            'agnosis_webhook_replay',
864            __( 'This webhook request has already been processed.', 'agnosis' ),
865            [ 'status' => 403 ]
866            );
867        }
868        set_transient( $replay_key, 1, self::REPLAY_MEMORY_SECONDS );
