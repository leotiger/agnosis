20/**
21 * Settings → General → "Preset biography title" (`agnosis_biography_preset_title`,
22 * default '') lets an operator force every biography page to use one fixed
23 * title instead of the artist's own — regardless of how or where that title
24 * was last set: ApplicationBiography's admission-derived "About {name}"
25 * draft, PostCreator's email-subject-as-title on creation or a later
26 * bio@ resend, an artist's own edit on the approve-confirm form (routed
27 * through ReviewEndpoints::save()), or a front-end post-publish edit via
28 * Artist\ContentEditor. Left blank (the default), nothing here changes
29 * anything and every one of those paths keeps working exactly as before —
30 * the artist's own title, whatever it is.
31 *
32 * Deliberately hooked on WordPress core's own `wp_insert_post_data` filter
33 * rather than any one of those individual write paths. That filter runs
34 * inside every single `wp_insert_post()`/`wp_update_post()` call — insert or
35 * update alike — immediately before the row is written, regardless of which
36 * of the paths above (or any future one) triggered it. This is one choke
37 * point instead of separately patching every title-setting call site above,
38 * which would be easy to leave inconsistent the next time a new one is added.
39 *
40 * That includes Lingua Forge's own translated-sibling creation and Agnosis's
41 * own native-language sibling creation (Compat\LinguaForge::sync_native_sibling()),
42 * so every biography page — every language — gets the preset title rather
43 * than the artist's own. But unlike the artwork dual-title system, a
44 * biography title is normally machine-translated per language (see this
45 * class's own translate_for_sibling(), and Compat\LinguaForge::hold_artist_title()'s
46 * docblock for why artwork is the one exception) — so a preset title should
47 * be too, not shown as the same fixed (likely English) text on every single
48 * language version. apply_preset_title() below still sets the untranslated
49 * preset as EVERY new post's initial title (simplest correct default, and
50 * exactly right for the primary-language post, which needs no translation at
51 * all) — translate_for_sibling(), hooked separately on
52 * `linguaforge_translation_complete` (fires for both a genuine LF-AI-translated
53 * sibling and Agnosis's own native-language one), then overwrites that initial
54 * value with a translated one for every OTHER language, the same "set once,
55 * correct for sibling afterward" shape sync_native_sibling() itself already
56 * uses for tags/medium.
57 *
58 * Settings → General → "Include artist's name in preset title"
59 * (`agnosis_biography_preset_title_include_name`, default off) appends the
60 * post author's own WordPress display name to the preset title when it's
61 * active — applied AFTER translation for a sibling, so the artist's name
62 * itself (a proper noun) is never run through the AI translator. Has no
63 * effect while the preset title itself is blank.
64 *

66 * form, front-end ContentEditor): while a preset title is configured, that
67 * field remains visible and editable but has no visible effect — this class
68 * overrides it again on the very next save, the same as it overrides
69 * whatever those forms pre-filled the field with. This is intentional: the

180        if ( 'agnosis_biography' !== get_post_type( $source_id ) ) {
181            return;
182        }
183
184        $preset_title = trim( (string) get_option( 'agnosis_biography_preset_title', '' ) );
185        if ( '' === $preset_title ) {
186            return;
187        }
188
189        $translated = self::translated_preset_title( $preset_title, $target_lang );
190        $final      = self::maybe_append_artist_name( $translated, (int) get_post_field( 'post_author', $translated_id ) );
191
192        // try/finally: a throwing wp_update_post() (or a callback it triggers)
193        // must never leave this flag stuck true — that would silently disable

205    /**
206     * Translate the preset title into $target_lang, cached per (text, language)
207     * so re-syncing the same sibling (a resubmission, a second staged update)
208     * never re-spends an AI call translating text that never changes — same
209     * shape and rationale as Compat\LinguaForge's own term-translation cache.
210     */
211    private static function translated_preset_title( string $preset_title, string $target_lang ): string {
212        $cache  = get_option( self::PRESET_TITLE_TRANSLATIONS_OPTION, [] );
213        $cached = $cache[ $preset_title ][ $target_lang ] ?? '';
214        if ( '' !== $cached ) {
215            return $cached;
216        }
217
218        $translator = SubmissionTranslator::from_settings();
219        if ( null === $translator ) {
220            return $preset_title;
221        }
222
223        $translated = trim( $translator->translate_text( $preset_title, $target_lang ) );
224        if ( '' === $translated ) {
225            return $preset_title;
226        }
227
228        $cache[ $preset_title ][ $target_lang ] = $translated;
229        update_option( self::PRESET_TITLE_TRANSLATIONS_OPTION, $cache, false );

231        return $translated;
232    }
