260        $value = get_option( 'agnosis_join_success_url', '' );
261
262        if ( ! is_numeric( $value ) ) {

264            // became a page selector — not a page ID, nothing to translate.
265            return (string) $value;
266        }
267
268        $page_id = (int) $value;
269        if ( $page_id <= 0 ) {

272
273        if ( '' !== $lang && LinguaForge::is_active() && function_exists( 'linguaforge_get_translations' ) ) {

280                }
281            }
282        }

284        $permalink = get_permalink( $page_id );
