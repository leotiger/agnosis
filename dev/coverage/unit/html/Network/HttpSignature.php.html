2/**
3 * HTTP Signature verification for incoming ActivityPub requests.
4 *
5 * Implements the subset of the HTTP Signatures draft spec used by Mastodon,
6 * Pixelfed, and the broader Fediverse:
7 *
8 *   • Signature header fields: keyId, headers, signature (algorithm ignored —
9 *     RSA-SHA256 is assumed, which all major Fediverse servers use).
10 *   • Signed headers supported: (request-target), host, date, digest.
11 *   • Body integrity: Digest: SHA-256=<base64> verified when included.
12 *   • Replay protection: Date header must be within ±12 hours of server time.
13 *   • Key caching: remote actor document is fetched once and cached for 24 h
14 *     in a transient, so repeated requests from the same actor cost one HTTP
15 *     round-trip per day rather than one per request. Fetch FAILURES are also
16 *     cached, separately and for 5 minutes, so a flood of requests bearing
17 *     random/nonexistent keyIds can't force one outbound fetch per request
18 *     (audit §3g note ii).
19 *   • keyId↔actor binding: verify_actor_binding() ties the signing key's
20 *     owner to the actor claimed inside the activity JSON (audit §3b).
21 *   • verify_with_key() generalizes the same signature check for callers
22 *     that already hold the caller-supplied public key inline (e.g. a peer
23 *     node registration, audit §2d), instead of fetching it from a remote
24 *     actor document.
25 *
26 * @package Agnosis\Network
27 */

115    /**
116     * Verify that the signing key belongs to the actor named in the activity.
117     *
118     * verify() proves the request was signed by the key at `keyId` — but
119     * nothing else ties that identity to the `actor` field inside the
120     * activity JSON the inbox then acts on. Without this binding, any
121     * fediverse account holder with a valid key on ANY server can send a
122     * correctly-signed request whose body claims someone else's actor id:
123     * a forged Follow subscribes the victim's inbox to every broadcast, and
124     * a forged Undo silently removes a real follower (audit §3b). Mastodon
125     * and friends guard exactly this by requiring the signing key's owner to
126     * match the activity's actor; this mirrors that rule.
127     *
128     * The keyId's base URL (fragment stripped) IS the actor document URL in
129     * every major implementation — Mastodon, Pleroma/Akkoma, GoToSocial, and
130     * Pixelfed all mint keyId as `<actor>#main-key` — so an exact match
131     * against the activity's actor id is both the simplest and the strictest
132     * correct comparison (the audit's "same-origin at minimum" floor would
133     * still allow same-server forgery; exact match closes that too).
134     *
135     * @param WP_REST_Request $request Incoming, already signature-verified request.

171    /**
172     * Shared steps 1–3 of signature verification: parse the Signature header,
173     * check date freshness, and verify the body digest when signed. Common to
174     * both verify() (fetches the key from keyId) and verify_with_key() (takes
175     * the key from the caller) — everything before "where does the public key
176     * come from" is identical between the two.
177     *
178     * @param WP_REST_Request $request Incoming request.
179     * @return array<string, string>|WP_Error Parsed Signature-header params on success.

182        if ( ! function_exists( 'openssl_verify' ) ) {
183            return new WP_Error(
184                'ap_sig_openssl_missing',
185                __( 'OpenSSL is required for HTTP Signature verification.', 'agnosis' ),
186                [ 'status' => 501 ]
187            );
188        }
189
190        // ── 1. Parse the Signature header ─────────────────────────────────────
191        $sig_header = (string) $request->get_header( 'signature' );
192        if ( '' === $sig_header ) {
193            return new WP_Error(
194                'ap_sig_missing',
195                __( 'Signature header is required.', 'agnosis' ),
196                [ 'status' => 401 ]
197            );
198        }
199
200        $params = self::parse_signature_header( $sig_header );
201        if ( ! isset( $params['keyId'], $params['headers'], $params['signature'] ) ) {

204                __( 'Signature header is missing required fields (keyId, headers, signature).', 'agnosis' ),
205                [ 'status' => 400 ]

210        // A signature legitimately minted over only "(request-target) host"
211        // verifies cryptographically fine — but it silently defeats the
212        // freshness check below (which only runs when a Date header happens to
213        // be present) and the digest check in verify_preamble()'s step 3 (which

218        // digest, since that's the request class carrying a body worth binding

224                __( 'Signature must cover the Date header.', 'agnosis' ),
225                [ 'status' => 401 ]
226            );
227        }
228        if ( 'POST' === $request->get_method() && ! str_contains( $signed_headers, 'digest' ) ) {
229            return new WP_Error(

231                __( 'Signature on a POST request must cover the Digest header.', 'agnosis' ),
232                [ 'status' => 401 ]

238        // is itself a rejection rather than a silent skip — previously a

353        }

356        $cached    = get_transient( $cache_key );

359        }
