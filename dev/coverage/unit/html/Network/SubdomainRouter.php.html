2/**
3 * Subdomain Router — artist-scoped subdomain routing.
4 *
5 * Maps wildcard subdomains to individual artist users so that
6 * artistx.agnosis.art serves only that artist's content.
7 *
8 * Architecture
9 * ────────────
10 * • DNS: wildcard *.{base_domain} → same WP install.
11 * • This class detects the subdomain on `plugins_loaded`, resolves it to a
12 *   WP user (nicename preferred, login as fallback), and wires query + URL
13 *   filters for the rest of the request.
14 * • `option_home` is rewritten to the artist subdomain so every
15 *   WP-generated URL (pagination, canonical, feeds, REST links…) is correct.
16 * • `pre_get_posts` scopes the main query to the artist's posts only.
17 *
18 * URL structure
19 * ─────────────
20 * artistx.agnosis.art            → artist's content, default language
21 * artistx.agnosis.art/en/        → English (LinguaForge subfolder mode)
22 * artistx.agnosis.art/ca/        → Catalan
23 * agnosis.art                    → portal / directory (no routing applied)
24 *
25 * @package Agnosis\Network
26 */

204    /**
205     * Build the subdomain URL for a given artist user.
206     *
207     * Prefers `user_nicename` (already URL-safe) over `user_login`.
208     * Falls back to `home_url()` when no base domain is configured.
209     *
210     * Carries the CURRENT request's language (via
211     * `Compat\LinguaForge::current_lang_path_prefix()`) so that following this
212     * link from a translated page (e.g. a visitor on `/fr/`) lands on the
213     * artist's `/fr/` home rather than always dropping back to their
214     * source-language root — '' when LF isn't active or the current language
215     * already IS the source language, so the common case is unaffected.
216     */
217    public static function url_for_artist( int $user_id ): string {
218        $base = (string) get_option( 'agnosis_base_domain', '' );
219        if ( ! $base ) {
220            return home_url();
221        }
222
223        $user = get_user_by( 'id', $user_id );
224        if ( ! $user ) {
225            return home_url();
226        }
227
228        $slug   = $user->user_nicename ?: $user->user_login;
229        $scheme = is_ssl() ? 'https' : 'http';
230        $prefix = \Agnosis\Compat\LinguaForge::current_lang_path_prefix();
231
232        // Trailing slash only when a language prefix is actually appended — the
233        // bare-domain case matches WordPress's own home_url() convention (no
234        // trailing slash), while '/fr' must become '/fr/' to be a well-formed
235        // terminal URL and avoid a needless 301 to the slashed form.
236        return $scheme . '://' . $slug . '.' . $base . $prefix . ( '' !== $prefix ? '/' : '' );
237    }
