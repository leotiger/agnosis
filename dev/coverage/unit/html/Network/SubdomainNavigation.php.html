505    /**
506     * Build the language breadcrumb badge: the artist's native language as a
507     * plain two-letter ISO 639-1 code (e.g. "ES"), with the language's own
508     * native name — its endonym, e.g. "Español", "Deutsch", "日本語" — as the
509     * hover tooltip and screen-reader label. Unlike the biography/events/
510     * contact variants, this renders no `<a href>`/`<button>` at all — there's
511     * nowhere for it to link to, it's purely informational.
512     *
513     * The native name comes from `SubmissionTranslator::language_names()`,
514     * which is sourced from Lingua Forge's `linguaforge_language_label()` —
515     * that function deliberately renders a language's name IN ITS OWN
516     * language (via `locale_get_display_language( $lang, $lang )`, same
517     * locale on both sides), not translated into the current visitor's own
518     * page language. That's exactly the behavior wanted here: "ES" always
519     * tooltips as "Español", regardless of what language the visitor is
520     * currently browsing the site in.
521     *
522     * Language source: the WP 'locale' user meta already set on the artist's
523     * own account at admission time, from the language they picked on the
524     * join form (`Admission::iso_to_wp_locale()` maps the submitted ISO code
525     * to a WP locale string like 'es_ES' before storing it). Reusing this
526     * existing meta — rather than adding a dedicated field — means an artist
527     * admitted before this meta was introduced (pre-0.9.22), or any account
528     * created outside the join flow, has no locale set and renders nothing;
529     * it also means an artist who later changes their own wp-admin interface
530     * language (Users → Profile → Language writes this same meta key) changes
531     * what this badge shows too — an accepted tradeoff for not needing a
532     * second, separately-maintained "native language" field.
533     *
534     * The first two characters of the locale reconstruct the original ISO
535     * code correctly for every entry in `iso_to_wp_locale()`'s map (every
536     * value there either IS the bare ISO code, e.g. 'fi', 'ar', or starts
537     * with it followed by an underscore, e.g. 'es_ES') — except 'zh-tw',
538     * which maps to 'zh_TW' and reduces back to just 'zh', same as plain
539     * Chinese. That's an accepted loss of precision for a two-letter badge —
540     * exactly what was asked for — not a bug.
541     */
542    private function render_language_badge( int $artist_id ): string {
543        $locale = (string) get_user_meta( $artist_id, 'locale', true );
544        $code   = strtolower( substr( $locale, 0, 2 ) );
545
546        if ( '' === $code ) {
547            return '';
548        }
549
550        $native_name = SubmissionTranslator::language_names()[ $code ] ?? '';
551
552        $this->enqueue_breadcrumb_icon_link_assets();
553
554        $extra_attributes = [ 'class' => 'agnosis-breadcrumb-icon-link--language' ];
555        if ( '' !== $native_name ) {
556            $extra_attributes['title']      = $native_name;
557            $extra_attributes['aria-label'] = $native_name;
558        }
559
560        $wrapper_attributes = get_block_wrapper_attributes( $extra_attributes );
561
562        return sprintf( '<span %1$s>%2$s</span>', $wrapper_attributes, esc_html( strtoupper( $code ) ) );
563    }

673     * language once publish happens, so an artist can have several published
674     * biography posts. Without scoping to the source language first,
675     * `get_posts()` could return ANY of those siblings (ordered by date, not
676     * language) — which is exactly why this link used to always point at
677     * whichever translation happened to be created last, regardless of which
678     * language the visitor was actually reading. `localized_post()` then
679     * swaps in the visitor's own language's sibling when one exists,
680     * mirroring `Newsletter\Digest::localized_post()`'s exact fallback chain.
681     */
682    private function biography_permalink( int $artist_id ): string {

684            'post_type'      => 'agnosis_biography',
685            'author'         => $artist_id,
686            'post_status'    => 'publish',
687            'posts_per_page' => 1,
688            'fields'         => 'ids',
689            'no_found_rows'  => true,

691
692        // phpcs:ignore WordPress.NamingConventions.PrefixAllGlobals.NonPrefixedFunctionFound -- Lingua Forge's public API; prefix belongs to that plugin.
693        if ( function_exists( 'linguaforge_source_language' ) ) {
694            $query_args['meta_query'] = [ // phpcs:ignore WordPress.DB.SlowDBQuery.slow_db_query_meta_query -- single, cheap lookup, once per artist-breadcrumb render.
695                'relation' => 'OR',
696                [ 'key' => '_lf_lang', 'value' => linguaforge_source_language() ],
697                [ 'key' => '_lf_lang', 'compare' => 'NOT EXISTS' ],

709    }

711    /**
712     * Resolve a post to its published translated counterpart in the CURRENT
713     * request's language (`LF_LANG`), or the post itself when Lingua Forge
714     * isn't active, the current language IS the source language, or no
715     * published translation exists yet.
716     */
717    private function localized_post( \WP_Post $post ): \WP_Post {
718        if ( ! defined( 'LF_LANG' ) || ! LF_LANG || ! function_exists( 'linguaforge_get_translations' ) ) {
719            return $post;
720        }
721
722        // phpcs:ignore WordPress.NamingConventions.PrefixAllGlobals.NonPrefixedFunctionFound -- Lingua Forge's public API; prefix belongs to that plugin.
723        $source = function_exists( 'linguaforge_source_language' ) ? linguaforge_source_language() : '';
724        if ( LF_LANG === $source ) {
725            return $post; // Already the source-language post — nothing to look up.
726        }
727
728        // phpcs:ignore WordPress.NamingConventions.PrefixAllGlobals.NonPrefixedFunctionFound -- Lingua Forge's public API; prefix belongs to that plugin.
729        $translations  = linguaforge_get_translations( $post->ID );
730        $translated_id = (int) ( $translations[ LF_LANG ] ?? 0 );
731        if ( $translated_id <= 0 ) {
732            return $post; // No translation into the visitor's language yet.
733        }
734
735        $translated = get_post( $translated_id );
736
737        return ( $translated instanceof \WP_Post && 'publish' === $translated->post_status ) ? $translated : $post;
738    }
