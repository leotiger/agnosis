2/**
3 * Lingua Forge compatibility layer.
4 *
5 * Integrates Agnosis with Lingua Forge (the official translation plugin for
6 * the Agnosis network) when both plugins are active on the same site.
7 *
8 * What this does when Lingua Forge is present:
9 *
10 *  1. LANGUAGE META — tags every new artwork post with `_lf_lang` (the source
11 *     language of the artist's submission) so LF's router, hreflang system,
12 *     and translation engine can all read the canonical language meta key.
13 *
14 *  2. TRANSLATION TRIGGER — after an artwork is published, calls LF's
15 *     `linguaforge_trigger_translation()` function once per target language
16 *     so the title, excerpt and body are translated into all configured site
17 *     languages without the artist doing anything.
18 *
19 *  3. SEO METADATA — overrides the Open Graph image with the artwork's
20 *     featured thumbnail via the `linguaforge_seo_og_image` filter.
21 *
22 *  4. ARTWORK SCHEMA — hooks into LF's `linguaforge_seo_schema_data` filter
23 *     to annotate artwork posts as `VisualArtwork` rather than generic `Article`.
24 *
25 * When Lingua Forge is NOT active, this class does nothing — all hooks are
26 * registered conditionally. No hard dependency.
27 *
28 * @package Agnosis\Compat
29 */

202    /**
203     * Schedule the deferred translation fan-out for a post, optionally excluding
204     * one or more target languages.
205     *
206     * Static and self-contained (touches no instance state) so callers outside the
207     * single `new LinguaForge()` instance Plugin.php constructs — e.g.
208     * Artist\ContentEditor after a front-end correction — can trigger a fan-out
209     * without instantiating a second LinguaForge object, which would re-register
210     * every constructor hook a second time (double SEO filters, double
211     * set_language_meta, etc.).
212     *
213     * Used two ways:
214     *   - schedule_translations() calls this with an empty exclusion list on every
215     *     normal publish.
216     *   - Artist\ContentEditor calls this directly after translating a front-end
217     *     correction into the primary language, excluding the artist's own source
218     *     language — that post already holds the artist's verbatim edit, and
219     *     re-deriving it via a second AI round-trip (primary -> artist's language
220     *     again) could drift from what the artist actually wrote (audit §7c,
221     *     reassessed 2026-07-06).
222     *
223     * @param int      $post_id       Source (primary-language) post ID.
224     * @param string[] $exclude_langs Target language codes to skip.
225     */
226    public static function schedule_fanout( int $post_id, array $exclude_langs = [] ): void {
227        if ( ! in_array( get_post_type( $post_id ), self::AGNOSIS_POST_TYPES, true ) ) {
228            return;
229        }
230
231        // Preserve the exact single-arg scheduling shape used since before Phase 1
232        // front-end correction existed when there is nothing to exclude — the args
233        // array is part of wp_next_scheduled()'s identity match, so appending an
234        // always-present (even if empty) second element would silently stop
235        // matching every pre-existing 1-arg wp_next_scheduled( DISPATCH_HOOK,
236        // [ $post_id ] ) lookup, including this class's own dedup check on the
237        // normal publish path and existing tests pinned to that signature.
238        $args = empty( $exclude_langs ) ? [ $post_id ] : [ $post_id, $exclude_langs ];
239
240        if ( ! wp_next_scheduled( self::DISPATCH_HOOK, $args ) ) {
241            wp_schedule_single_event( time(), self::DISPATCH_HOOK, $args );
242        }
243    }

245    /**

247     *
248     * Order matters and is race-free: the per-language title map is built first
249     * (synchronously within this cron tick), then the body translations are queued.
250     * The queued translation jobs run in still-later ticks, by which point the map
251     * already exists for supply_translated_meta() to read.
252     *
253     * @param int      $post_id       Source post ID.

261    /**
262     * Ask Lingua Forge to translate the post into all configured languages.
263     *
264     * Prefers `linguaforge_queue_translation()` (LF 2.4.0+) so each translation runs
265     * off-request via Action Scheduler / WP-Cron; falls back to the synchronous
266     * `linguaforge_trigger_translation()` on older LF. LF translates the title,
267     * content and excerpt and creates a TRID-linked post per language; images, the
268     * per-language display title, and `_lf_lang` are handled by the hooks above.
269     *
270     * No `$params` are passed: LF's defaults are exactly what we want. (The plugin
271     * previously sent `domain` / `priority` / `source` — none of which LF reads, so
272     * they were silently dropped. `with_meta_description` is intentionally not set:
273     * the translated excerpt LF already produces is the artwork's description.)
274     *
275     * Returns silently when LF is not loaded or there are no target languages.
276     *
277     * @param int      $post_id       Source post ID.
278     * @param string[] $exclude_langs Target language codes to skip in addition to
279     *                                the source language (already excluded by
280     *                                get_target_languages()). See schedule_fanout().
281     */
282    public function request_translations( int $post_id, array $exclude_langs = [] ): void {
283        if ( ! in_array( get_post_type( $post_id ), self::AGNOSIS_POST_TYPES, true ) ) {
284            return;
285        }
286
287        if ( ! function_exists( 'linguaforge_trigger_translation' ) ) {
288            return;
289        }
290
291        $source_lang = get_post_meta( $post_id, '_lf_lang', true ) ?: 'en';
292        $languages   = $this->get_target_languages( $source_lang );
293
294        if ( ! empty( $exclude_langs ) ) {
295            $languages = array_values( array_diff( $languages, $exclude_langs ) );
296        }
297
298        if ( empty( $languages ) ) {
299            return;
300        }
301
302        foreach ( $languages as $target_lang ) {
303            // Prefer the async queue (LF 2.4.0+); fall back to the synchronous trigger.
304            // function_exists() is checked inline so static analysis narrows correctly.
305            if ( function_exists( 'linguaforge_queue_translation' ) ) {
306                // phpcs:ignore WordPress.NamingConventions.PrefixAllGlobals.NonPrefixedFunctionFound -- calling Lingua Forge's public API; prefix belongs to that plugin.
