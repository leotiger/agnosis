2/**
3 * Lingua Forge compatibility layer.
4 *
5 * Integrates Agnosis with Lingua Forge (the official translation plugin for
6 * the Agnosis network) when both plugins are active on the same site.
7 *
8 * What this does when Lingua Forge is present — eight concerns, registered in
9 * the constructor in this order:
10 *
11 *  1. LANGUAGE META — tags every new artwork post with `_lf_lang` (the source
12 *     language of the artist's submission) so LF's router, hreflang system,
13 *     and translation engine can all read the canonical language meta key.
14 *
15 *  2. TRANSLATION TRIGGER — after an artwork is published, calls LF's
16 *     `linguaforge_trigger_translation()` function once per target language
17 *     so the title, excerpt and body are translated into all configured site
18 *     languages without the artist doing anything.
19 *
20 *  3. DUAL-TITLE HANDLING — keeps the artist's own `post_title` untranslated
21 *     on every language version (via `linguaforge_translation_content`,
22 *     hold_artist_title()); the per-language display title AI-translates
23 *     separately into `_agnosis_translated_title`. Artwork and event
24 *     (DUAL_TITLE_POST_TYPES) — biography's own title still uses LF's
25 *     normal per-sibling translation instead (see hold_artist_title()'s
26 *     docblock for why).
27 *
28 *  4. TRANSLATED-POST META PROPAGATION — a translated post is otherwise
29 *     created (and, critically, re-translated) with none of the source's
30 *     images/event/gallery meta; copy_translated_meta() / supply_translated_meta()
31 *     copy a language-neutral allowlist across, refreshing it again on every
32 *     re-translation (fourth audit, §4b).
33 *
34 *  5. TAG / MEDIUM TRANSLATION — LF's own translation pipeline never touches
35 *     taxonomy terms at all, so a translated post is otherwise created with no
36 *     tags and no medium term whatsoever. sync_translated_terms() translates
37 *     and assigns both onto every translated sibling, cached per (taxonomy,
38 *     term name, language) so a recurring value (a common tag, or one of
39 *     `agnosis_medium`'s built-in options) gets the same translated label
40 *     every time rather than a fresh AI phrasing per post. Newly-created
41 *     translated terms are flagged with TRANSLATED_TERM_META so the AI's own
42 *     controlled vocabulary (`PromptConfig::medium_terms()`) doesn't end up
43 *     polluted with machine-translated variants (fourth audit, §4c). The
44 *     cache itself can be cleared from Settings → General (a manual "Clear
45 *     Term Translation Cache" action) and is auto-invalidated per entry when
46 *     a source term is renamed, since the cache key is the term's name
47 *     (fourth audit, §4d).
48 *
49 *  6. SEO METADATA — overrides the Open Graph image with the artwork's
50 *     featured thumbnail via the `linguaforge_seo_og_image` filter.
51 *
52 *  7. ARTWORK SCHEMA — hooks into LF's `linguaforge_seo_schema_data` filter
53 *     to annotate artwork posts as `VisualArtwork` rather than generic `Article`.
54 *
55 *  8. TEMPLATE SAFEGUARD — LF >= 2.6.1 only. LF 2.6.1 fixed
56 *     `TranslationTrigger::create_translated_post()` (the function Agnosis's
57 *     own translation trigger above always goes through) so a newly created
58 *     translation is correctly assigned its language-specific FSE template
59 *     (`single-{post_type}-{lang}`) when one already exists in the DB. This
60 *     is a defense-in-depth call on top of that fix, not a workaround for its
61 *     absence: `sync_translated_template()` calls LF's new
62 *     `linguaforge_sync_templates()` (also 2.6.1) after every translation
63 *     completes, which re-resolves and re-writes `_wp_page_template` for
64 *     every sibling in the post's translation group. It makes no AI call and
65 *     touches no content — LF's own changelog documents it as "free to run
66 *     and safe to run repeatedly" — so keeping it as a standing safeguard
67 *     costs nothing even now that the underlying bug is fixed, and continues
68 *     to protect against template drift from a future theme change, template
69 *     rename, or an LF regression, without Agnosis needing to know any of
70 *     LF's own template-resolution logic itself. No-ops entirely on any LF
71 *     version before 2.6.1 (function_exists() guard) — the fix above still
72 *     applies on 2.6.1+ either way, this is additive only.
73 *
74 * Since 0.9.22, agnosis.php declares `Requires Plugins: lingua-forge` —
75 * WordPress itself now refuses to install or activate Agnosis at all until
76 * Lingua Forge is installed and active (and, symmetrically, refuses to let an
77 * operator deactivate or delete Lingua Forge while Agnosis is active). This
78 * class's own function_exists()/class_exists() gating on every hook
79 * registration stays regardless, as defense-in-depth: WordPress core's own
80 * guidance on Plugin Dependencies is explicit that this mechanism only
81 * enforces presence at install/activation time, has no minimum-version
82 * support, and doesn't account for plugin load order — it does not replace
83 * runtime feature-detection. Concretely, that means this class still no-ops
84 * safely rather than fatal-erroring if Lingua Forge is ever removed via FTP,
85 * or during the brief window before it's loaded relative to Agnosis on a
86 * given request.
87 *
88 * @package Agnosis\Compat
89 */

395    /**
396     * Schedule the deferred translation fan-out for a post, optionally excluding
397     * one or more target languages.
398     *
399     * Static and self-contained (touches no instance state) so callers outside the
400     * single `new LinguaForge()` instance Plugin.php constructs — e.g.
401     * Artist\ContentEditor after a front-end correction — can trigger a fan-out
402     * without instantiating a second LinguaForge object, which would re-register
403     * every constructor hook a second time (double SEO filters, double
404     * set_language_meta, etc.).
405     *
406     * Used two ways:
407     *   - schedule_translations() calls this with an empty exclusion list on every
408     *     normal publish.
409     *   - Artist\ContentEditor calls this directly after translating a front-end
410     *     correction into the primary language, excluding the artist's own source
411     *     language — that post already holds the artist's verbatim edit, and
412     *     re-deriving it via a second AI round-trip (primary -> artist's language
413     *     again) could drift from what the artist actually wrote (audit §7c,
414     *     reassessed 2026-07-06).
415     *
416     * @param int      $post_id       Source (primary-language) post ID.
417     * @param string[] $exclude_langs Target language codes to skip.
418     */
419    public static function schedule_fanout( int $post_id, array $exclude_langs = [] ): void {
420        if ( ! in_array( get_post_type( $post_id ), self::AGNOSIS_POST_TYPES, true ) ) {
421            return;
422        }
423
424        // Preserve the exact single-arg scheduling shape used since before Phase 1
425        // front-end correction existed when there is nothing to exclude — the args
426        // array is part of wp_next_scheduled()'s identity match, so appending an
427        // always-present (even if empty) second element would silently stop
428        // matching every pre-existing 1-arg wp_next_scheduled( DISPATCH_HOOK,
429        // [ $post_id ] ) lookup, including this class's own dedup check on the
430        // normal publish path and existing tests pinned to that signature.
431        $args = empty( $exclude_langs ) ? [ $post_id ] : [ $post_id, $exclude_langs ];
432
433        if ( ! wp_next_scheduled( self::DISPATCH_HOOK, $args ) ) {
434            wp_schedule_single_event( time(), self::DISPATCH_HOOK, $args );
435        }
436    }

443     * The queued translation jobs run in still-later ticks, by which point the map

445     *

454    /**
455     * Ask Lingua Forge to translate the post into all configured languages.
456     *
457     * Prefers `linguaforge_queue_translation()` (LF 2.4.0+) so each translation runs
458     * off-request via Action Scheduler / WP-Cron; falls back to the synchronous
459     * `linguaforge_trigger_translation()` on older LF. LF translates the title,
460     * content and excerpt and creates a TRID-linked post per language; images, the
461     * per-language display title, and `_lf_lang` are handled by the hooks above.
462     *
463     * No `$params` are passed: LF's defaults are exactly what we want. (The plugin
464     * previously sent `domain` / `priority` / `source` — none of which LF reads, so
465     * they were silently dropped. `with_meta_description` is intentionally not set:
466     * the translated excerpt LF already produces is the artwork's description.)
467     *
468     * Returns silently when LF is not loaded or there are no target languages.
469     *
470     * @param int      $post_id       Source post ID.
471     * @param string[] $exclude_langs Target language codes to skip in addition to
472     *                                the source language (already excluded by
473     *                                get_target_languages()). See schedule_fanout().
474     */
475    public function request_translations( int $post_id, array $exclude_langs = [] ): void {
476        if ( ! in_array( get_post_type( $post_id ), self::AGNOSIS_POST_TYPES, true ) ) {
477            return;
478        }
479
480        if ( ! function_exists( 'linguaforge_trigger_translation' ) ) {
481            return;
482        }
483
484        $source_lang = get_post_meta( $post_id, '_lf_lang', true ) ?: 'en';
485        $languages   = $this->get_target_languages( $source_lang );
486
487        if ( ! empty( $exclude_langs ) ) {
488            $languages = array_values( array_diff( $languages, $exclude_langs ) );
489        }
490
491        if ( empty( $languages ) ) {
492            return;
493        }
494
495        foreach ( $languages as $target_lang ) {
496            // Prefer the async queue (LF 2.4.0+); fall back to the synchronous trigger.
497            // function_exists() is checked inline so static analysis narrows correctly.
498            if ( function_exists( 'linguaforge_queue_translation' ) ) {
499                // phpcs:ignore WordPress.NamingConventions.PrefixAllGlobals.NonPrefixedFunctionFound -- calling Lingua Forge's public API; prefix belongs to that plugin.
500                linguaforge_queue_translation( $post_id, $target_lang );
501            } else {
502                // phpcs:ignore WordPress.NamingConventions.PrefixAllGlobals.NonPrefixedFunctionFound -- calling Lingua Forge's public API; prefix belongs to that plugin.
503                linguaforge_trigger_translation( $post_id, $target_lang );
504            }
505        }
506    }

586            return; // Site isn't configured to route to this language at all.
587        }

589        $primary_lang = (string) get_post_meta( $primary_post_id, '_lf_lang', true );
590        if ( $native_lang === $primary_lang ) {
591            return; // Artist already writes in the site's primary language — the primary post IS the native one.
592        }

595        $native_body    = (string) get_post_meta( $primary_post_id, '_agnosis_native_body', true );

601        if ( ! $source instanceof \WP_Post ) {
602            return;
603        }

606            || ! function_exists( 'linguaforge_get_translations' ) || ! function_exists( 'linguaforge_clear_translation_cache' )

675        // try/finally (seventh audit §2c): without it, a throwing listener on

697        }

700            $native_medium = (string) get_post_meta( $primary_post_id, '_agnosis_native_medium', true );
701            if ( '' !== $native_medium ) {

703            }
704        }

708     * Trash the native-language sibling built for $old_native_lang, when a
709     * staged update changes (or clears) the artist's declared native

732        $translations = linguaforge_get_translations( $primary_post_id );
733        $sibling_id   = (int) ( $translations[ $old_native_lang ] ?? 0 );
734
735        // No sibling ever existed for that language (e.g. the artist's prior
736        // language wasn't LF-configured, or sync_native_sibling() had
737        // already no-op'd for some other reason) — nothing to trash. The

742            return;
743        }

748        }

755                $sibling_id,
756                $old_native_lang,
757                $primary_post_id
758            ),
759            'lingua-forge'
760        );

765     * sync_native_sibling(). Mirrors `TranslationTrigger::create_translated_post()`'s
766     * own recipe: bypass `wp_after_insert_post` handlers during the insert
767     * (same reasoning LF's own code documents — those handlers assume a

775     *
776     * post_title is the primary post's own post_title, verbatim — never a
777     * synthetic "Title [XX]" fallback the way LF's own `build_create_args()`
778     * falls back to for an AI translation with no result: this class's
779     * dual-title design means post_title is ALREADY the artist's own native
780     * words, identical for every language sibling by design (see
781     * hold_artist_title()) — there's nothing to translate or fall back on here.

788            'post_excerpt' => $excerpt,
789            'post_content' => $content,
790            'post_status'  => $source->post_status,
791            'post_type'    => $source->post_type,

795        // See TranslationTrigger::create_translated_post()'s own docblock for
796        // this filter's contract — same filter, same shape.
797        $meta = (array) apply_filters(
798            'linguaforge_translated_post_meta', // phpcs:ignore WordPress.NamingConventions.PrefixAllGlobals.NonPrefixedHooknameFound -- linguaforge_ is the registered plugin prefix.
799            [],
800            $source->ID,
801            $target_lang,
802            $source->post_type
803        );

824            );

826        }

829        update_post_meta( $new_id, '_lf_lang', $target_lang );

831        return (int) $new_id;
832    }
