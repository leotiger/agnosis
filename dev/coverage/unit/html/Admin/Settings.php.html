321    /** @param array<string, mixed> $field */

323        $value = get_option( $key, $field['default'] ?? '' );
324        $type  = $field['input'] ?? 'text';

328
329        switch ( $type ) {
330            case 'password':
331                echo '<input type="password" id="' . esc_attr( $key ) . '" name="' . esc_attr( $key ) . '" value="' . esc_attr( $value ) . '" class="regular-text" autocomplete="new-password">';

338                break;
339            case 'select':
