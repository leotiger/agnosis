2/**
3 * SubmissionTranslator — normalise artist email submissions to the site's primary language.
4 *
5 * Artists submit their work by email in their native language (Spanish, German,
6 * Chinese, etc.). Before the AI pipeline processes their text (subject + body),
7 * it must be in the site's primary language so that title, excerpt, body, and
8 * tags are published in the correct language.
9 *
10 * Target language resolution order (mirrors Lingua Forge's own resolution):
11 *   1. `linguaforge_primary_language` WordPress option — set in Lingua Forge →
12 *      Settings → Primary Language.
13 *   2. WordPress site locale (first two characters of `get_locale()`), so an
14 *      unconfigured install still behaves sensibly.
15 *   3. `'en'` — hard fallback when the locale cannot be resolved to a known code.
16 *
17 * The translation is performed in a single `chat()` call with a JSON envelope
18 * so subject and body are translated together without two round trips. The
19 * envelope is stripped on parse. If translation fails (empty or non-JSON
20 * response) the original text is preserved — the pipeline always continues,
21 * just potentially with untranslated text.
22 *
23 * @package Agnosis\AI
24 */

175     * strings.

185     * prompt and the returned array — same convention `call_translate()`
186     * already uses — so callers should fall back to the original text for

188     * `translate_text()` call. An entirely failed/unparseable response
189     * returns an empty array; callers can distinguish "nothing needed
190     * translating" from "the call failed" by checking whether $fields was
191     * non-empty going in.

194     * @param string                $target_code ISO 639-1 code (e.g. 'es', 'fr', 'zh').
195     * @return array<string, string> Field name => translated text, only for
196     *                               fields that were non-empty AND present in
197     *                               the AI's response.

388     *
389     * Single source of truth for "what language does this artist write in" —
390     * used by the native-first AI pipeline (Pipeline::process(), instructing

392     * PostCreator::create_post() (persisting `_agnosis_native_lang` once at
393     * intake), and ReviewConfirm (display/back-translation decisions) — three
394     * call sites that previously risked drifting apart with their own copies

398     * callers should treat that the same as "language unknown," same
399     * graceful-degradation convention every other resolution method here uses.
400     */
401    public static function resolve_artist_lang( int $artist_id ): string {
402        if ( ! $artist_id ) {
403            return '';
404        }

406        if ( '' === $locale ) {

408        }

451    /**
452     * Send a single chat() call that translates both subject and body together.
453     *
454     * Returns an associative array with 'subject' and/or 'description' keys, or
455     * null on failure (empty response, non-JSON, or missing keys).
456     *
457     * The prompt includes both fields in one round trip. Fields that are empty
458     * in the input are omitted from the prompt and from the returned array so
459     * the caller knows not to overwrite them.
460     *
461     * @return array<string, string>|null
462     */
463    private function call_translate( string $subject, string $body, string $target_language_name ): ?array {
464        $sections = '';
465        if ( '' !== $subject ) {
466            $sections .= "SUBJECT:\n{$subject}\n\n";
467        }
468        if ( '' !== $body ) {
469            $sections .= "BODY:\n{$body}";
470        }
471
472        $keys_present = array_filter( [ 'subject' => $subject !== '', 'body' => $body !== '' ] );
473        $json_keys    = implode( ', ', array_map(
474            static fn( $k ) => '"' . ( $k === 'body' ? 'description' : $k ) . '"',
475            array_keys( $keys_present )
476        ) );
477
478        $prompt = "Translate the sections below to {$target_language_name}.\n"
479            . "If a section is already in {$target_language_name}, include it in the output unchanged.\n"
480            . "Return ONLY a JSON object with these keys: {$json_keys}.\n"
481            . "No markdown fences. No preamble. No explanation.\n\n"
482            . $sections;
483
484        $response = $this->provider->chat( $prompt );
485
486        if ( '' === trim( $response ) ) {
487            return null;
488        }
489
490        // Strip markdown fences if present.
491        $json_str = trim( (string) preg_replace( '/^```(?:json)?\s*|\s*```$/', '', trim( $response ) ) );
492        $decoded  = json_decode( $json_str, true );
493
494        if ( ! is_array( $decoded ) ) {
495            $this->log_json_decode_failure( 'call_translate', $json_str, $target_language_name, implode( '+', array_keys( $keys_present ) ) );
496            return null;
497        }
498
499        $result = [];
500
501        // is_string() guard (not a blind (string) cast): if the model's JSON
502        // response ever has 'subject'/'description' as a non-scalar (an array
503        // or nested object — seen in practice translating a single short word
504        // like a biography's preset title, "About", with no surrounding
505        // sentence context to anchor the model's response shape), casting
506        // an array to string doesn't fail loudly — it silently produces the
507        // literal string "Array" (with a PHP notice, easy to miss in
508        // production), which then gets published as real, user-visible
509        // content (e.g. "Array — Cal Talaia" surfaced 2026-07-13 via
510        // Artist\BiographyTitle::translate_for_sibling(), the first caller of
511        // translate_text() to feed it a single bare word). Treating a
512        // non-string response as a failed field — same as the missing-key
513        // case just below each check — means every caller's existing
514        // "fall back to the original text" convention (translate_text(),
515        // translate_fields(), the callers in Compat\LinguaForge) applies
516        // here too, instead of ever publishing "Array" as if it were a
517        // genuine translation.
518        if ( '' !== $subject && isset( $decoded['subject'] ) && is_string( $decoded['subject'] ) ) {
519            $result['subject'] = sanitize_text_field( $decoded['subject'] );
520        }
521
522        if ( '' !== $body && isset( $decoded['description'] ) && is_string( $decoded['description'] ) ) {
523            $result['description'] = sanitize_textarea_field( $decoded['description'] );
524        }
525
526        return ! empty( $result ) ? $result : null;
527    }

529    /**
530     * Log a JSON-decode failure with a distinct marker for the likely
531     * truncation case (audit §5b), so it surfaces in Settings → Logs as
532     * itself rather than as an unexplained, generic translation failure —
533     * previously none of the three JSON-envelope call sites above logged
534     * anything at all on decode failure, silently falling back to the
535     * original untranslated text.
536     *
537     * Distinguishing "truncated mid-response" from "the model returned
538     * something else entirely" doesn't require the provider's own
539     * finish_reason/stop_reason (threading that through would mean widening
540     * ProviderInterface::chat()'s string return type everywhere it's used,
541     * including Pipeline.php's own six call sites and every test double —
542     * far more invasive than this finding calls for). A cheap, reliable
543     * proxy instead: a response only counts as "likely truncated" when it
544     * actually looks like it STARTED as the requested JSON object (begins
545     * with "{") but doesn't END with the matching closing brace — cut off
546     * mid-object. Checking the closing brace alone isn't enough: a response
547     * that never was JSON in the first place (the model refused, or replied
548     * with plain prose) also won't end in "}", but that's a different
549     * failure than truncation and would be mislabeled without the opening-
550     * brace check too.
551     *
552     * @param string $method       Calling method name, for the log message.
553     * @param string $json_str     The fence-stripped string that failed to decode.
554     * @param string $target       Target language name/code(s), for context.
555     * @param string $field_summary Short description of what was being translated.
556     */
557    private function log_json_decode_failure( string $method, string $json_str, string $target, string $field_summary ): void {
558        $likely_truncated = '' !== $json_str && '{' === $json_str[0] && '}' !== substr( $json_str, -1 );
559
560        Logger::warning(
561            sprintf(
562                'SubmissionTranslator::%s: AI response was not valid JSON (%s) — target "%s", %s. Falling back to the original untranslated text.',
563                $method,
564                $likely_truncated
565                    ? 'looks truncated — response does not end with a closing brace; consider whether max_tokens needs raising further'
566                    : 'malformed or unexpected response shape, not a truncation',
567                $target,
568                $field_summary
569            ),
570            'pipeline'
571        );
572    }
