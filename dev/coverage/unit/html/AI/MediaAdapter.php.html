2/**
3 * MediaAdapter — normalise non-image attachments into pipeline-compatible shapes.
4 *
5 * The Pipeline was originally designed for image/jpeg and image/png attachments.
6 * This adapter sits in front of the pipeline loop and converts other media types
7 * into a form the pipeline can process:
8 *
9 *   image/heic,       → converted to JPEG with Imagick (see adapt_heic()) — neither
10 *   image/heif          the vision AI providers nor most non-Safari browsers can read
11 *                        HEIC/HEIF directly, so unlike plain image/* this format needs
12 *                        an actual pixel conversion, not just a passthrough.
13 *   image/*          → passthrough with media_type = 'image' (no conversion needed).
14 *   application/pdf  → rasterise each page with Imagick, emit one 'image' entry per page.
15 *   video/*          → extract the first frame with ffmpeg as a poster/description
16 *                       image, but keep the original video file — emit one 'video'
17 *                       entry carrying both. The video itself is what gets published;
18 *                       the frame is only used for AI description and the <video poster>.
19 *   audio/*          → passthrough with media_type = 'audio' (pipeline routes to audio branch).
20 *
21 * Attachments that cannot be converted (Imagick unavailable for PDFs or HEIC/HEIF —
22 * the latter also requires the libheif delegate, which isn't bundled with every
23 * ImageMagick build) are logged and dropped — the caller receives fewer entries than
24 * it passed in. Video is more forgiving: if ffmpeg is unavailable or frame extraction
25 * fails, the video itself is still forwarded (with no poster frame) rather than being
26 * dropped entirely — Pipeline falls back to a text-only description in that case, same
27 * as it does for audio with no transcript available.
28 *
29 * Each attachment array is expected to have at minimum:
30 *   'data'       string  Raw binary content.
31 *   'mime'       string  MIME type (matches Parser output).
32 *   'filename'   string  Original filename.
33 *
34 * Each returned entry carries those same keys plus:
35 *   'media_type'  string  'image' | 'audio' | 'video' — tells Pipeline which branch to use.
36 *   'mime'        string  Possibly updated MIME (e.g. 'image/jpeg' after PDF rasterisation).
37 *   'poster_data' string  (video only) Extracted first-frame JPEG binary, or '' if unavailable.
38 *   'poster_mime' string  (video only) Always 'image/jpeg' when poster_data is non-empty.
39 *
40 * Also home to maybe_downscale_for_vision() — a per-request image-downscale
41 * helper for vision API calls (cost optimisation), deliberately NOT wired
42 * into adapt()/adapt_pdf()/adapt_video() themselves. See that method's own
43 * doc for why: the attachment 'data'/'poster_data' this class returns is
44 * reused downstream for enhancement and for the actual published file, so
45 * mutating it here would silently shrink published artwork.
46 *
47 * @package Agnosis\AI
48 */

146     * Downscale an image to the admin-configured max width (agnosis_ai_vision_max_width_px,
147     * default 800px) for a single vision-API request — a pure cost/latency
148     * optimisation, since vision-token cost scales with resolution and a
149     * description task needs nowhere near a photo's original resolution.
150     *
151     * IMPORTANT — call this from inside each ProviderInterface::describe()
152     * implementation, immediately before base64-encoding the request body,
153     * and use only the return value for that one request. Never call it here
154     * in MediaAdapter (i.e. never let its output replace an attachment's
155     * 'data'/'poster_data') and never let its output flow back into
156     * Pipeline::process_single()'s $image_data / process_video_single()'s
157     * $poster_data. Those same variables are also used, unmodified, for:
158     * the image enhancement API call (which needs real resolution to do a
159     * good job), Pipeline's 'original_data' result (the actual file
160     * PostCreator publishes when enhancement doesn't run), and — for video —
161     * the poster frame's own 'poster_data' result (the real `<video poster>`
162     * attachment). Downscaling any of those in place would silently publish
163     * artwork/poster images at only agnosis_ai_vision_max_width_px wide,
164     * which is not what this setting is for.
165     *
166     * Deliberately conservative: never upscales (a source already narrower
167     * than the configured max is returned untouched), and any failure —
168     * disabled via the '0' setting, Imagick unavailable, or a corrupt/
169     * unreadable blob — falls back to returning the original image
170     * unmodified rather than ever blocking a describe() call. Mirrors
171     * adapt_pdf()'s existing Imagick-availability guard and logging
172     * convention, since Imagick is already a soft dependency in this class.
173     *
174     * @param string $data Raw image binary.
175     * @param string $mime Image MIME type (informational only — Imagick
176     *                     preserves the source format on resize).
177     * @return string Resized image binary, or the original $data unchanged
178     *                if downscaling is disabled, unnecessary, or unavailable.
179     */

183        if ( $max_width <= 0 || '' === $data ) {
184            return $data; // Disabled via the admin setting, or nothing to resize.

186

192            return $data;

297     * Convert a HEIC/HEIF photo to JPEG with Imagick.
298     *
299     * iPhones default to HEIC ("High Efficiency") for photos unless the artist
300     * switched Settings → Camera → Formats to "Most Compatible". Unlike the
301     * generic image/* passthrough in adapt(), this format can't just ride
302     * through unchanged: the vision AI providers (OpenAI/Anthropic accept
303     * JPEG/PNG/WebP/GIF, not HEIC/HEIF) and most non-Safari browsers can't
304     * read it, so a real pixel conversion has to happen before the file
305     * reaches either.
306     *
307     * This needs more than "Imagick is installed" (adapt_pdf()'s guard) —
308     * decoding HEIC/HEIF specifically requires the ImageMagick build to have
309     * been compiled with the libheif delegate, which plenty of hosts don't
310     * include (licensing). Rather than probing for the delegate separately,
311     * the read is simply attempted and any \ImagickException — missing
312     * delegate, corrupt file, anything else — is treated the same as Imagick
313     * being absent: log a specific reason and drop the attachment, exactly
314     * like adapt_pdf() does, rather than ever letting raw HEIC bytes through.
315     *
316     * A HEIC/HEIF container can hold a burst or Live Photo sequence; only the
317     * first (primary) image is used — the same "one artwork, one photo" shape
318     * every other image path in this pipeline assumes.
319     *
320     * @param array<string, mixed> $attachment  Raw attachment with 'data', 'mime', 'filename'.
321     * @return array<int, array<string, mixed>>  Zero or one entries — empty on any failure.
322     */

325            Logger::warning( sprintf(
326                'MediaAdapter: HEIC/HEIF "%s" skipped — Imagick PHP extension not available.',
327                $attachment['filename'] ?? 'unknown'
328            ), 'pipeline' );
329            return [];
330        }
331
332        try {
333            $imagick = new \Imagick();
334            $imagick->readImageBlob( $attachment['data'] );
335            $imagick->setIteratorIndex( 0 ); // Primary image only — ignore any burst/sequence frames.
336            $imagick->setImageFormat( 'jpeg' );
337            $imagick->setImageCompressionQuality( 90 );
338
339            $jpeg_data = $imagick->getImageBlob();
340            $imagick->destroy();
341
342            if ( ! $jpeg_data ) {
343                Logger::warning( sprintf(
344                    'MediaAdapter: HEIC/HEIF "%s" — Imagick produced no output, skipping.',
345                    $attachment['filename'] ?? 'unknown'
346                ), 'pipeline' );
347                return [];
348            }
349
350            $base_name = pathinfo( (string) ( $attachment['filename'] ?? 'photo' ), PATHINFO_FILENAME );
351
352            $entry = [
353                'data'       => $jpeg_data,
354                'mime'       => 'image/jpeg',
355                'filename'   => $base_name . '.jpg',
356                'media_type' => 'image',
357            ];

363            // without the libheif delegate, so it can list HEIC as a known
364            // format but can't actually decode one.
365            Logger::warning( sprintf(
366                'MediaAdapter: HEIC/HEIF "%s" skipped — this server\'s ImageMagick build could not decode it (%s). Likely missing the libheif delegate.',
367                $attachment['filename'] ?? 'unknown',
368                $e->getMessage()
369            ), 'pipeline' );
370            return [];
371        }
