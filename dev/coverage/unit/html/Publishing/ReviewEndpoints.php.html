606
607        return $pending_for;
608    }

614    /**
615     * Extract the leading image/gallery block markup from post content.
616     *

618     * block(s) at the top of the content — only the text below them changes.

625        if ( preg_match( '/^((?:<!-- wp:(?:image|gallery)[^>]*-->.*?<!-- \/wp:(?:image|gallery) -->[\s]*)+)/s', $content, $matches ) ) {
626            return trim( $matches[1] );
627        }
628        return '';
629    }

634     * publish — Phase 3 of the native-language pipeline redesign
635     * (agnosis-audit/NATIVE-LANGUAGE-PIPELINE.md §4c). Called from both
636     * branches of finalize_publish() so a staged update and a first-time
637     * publish are translated identically.

640     * live post — it stays the artist's own verbatim words at rest everywhere
641     * in this plugin (the dual-title design predates this feature; see
642     * PostCreator::create_post()'s '_agnosis_original_title' handling and this
643     * class's own callers, neither of which has ever translated post_title).
644     * The translated title this method DOES produce is a separate,
645     * display-only copy — 'display_title' in the return value — meant for
646     * '_agnosis_translated_title' (Compat\LinguaForge's dual-title system),

650     * trustworthy as a primary-language translation.

653     * fields unchanged" — when: $source has no declared native language

657     * common single-language case costs nothing extra, same convention every
658     * other translation method in this codebase uses), or the translation call
659     * itself fails (logged, falls back to publishing the native-language
660     * content as-is rather than blocking the approval entirely).
661     *
662     * Also returns the untranslated native excerpt/body ('native_excerpt'/
663     * 'native_body' below) — the design doc's Phase 2 (§4b, "hold the native
664     * result, don't discard it") — so the caller can persist them onto the
665     * post that actually survives (the target of a staged update, or $source
666     * itself for a first-time publish) BEFORE they're overwritten with the
667     * primary translation below. Deliberately captured HERE, at approval,
668     * rather than at intake as §4b originally proposed: this is the point the
669     * FINAL text is known — the artist's original AI-generated result, or
670     * their edit of it if they changed anything on the confirm form — so
671     * what's preserved is what was actually approved, not a possibly-stale
672     * intake-time snapshot. Without this, once this method's caller writes the
673     * primary translation over post_excerpt/post_content, the native-language
674     * version would be gone entirely — the one thing Phase 4 (creating the
675     * artist's own native-language sibling post, agnosis-audit/
676     * NATIVE-LANGUAGE-PIPELINE.md §4d) needs to exist at all.
677     *
678     * @return array{display_title: string, excerpt: string, content: string, medium: string, tags: string[], native_excerpt: string, native_body: string, native_tags: string[]}|null
679     */

736        // doc accounts for per cross-language approval. Recorded here, once

754            'content'        => $content,
755            'medium'         => trim( $translated['medium'] ?? $native_medium ),
756            'tags'           => $tags,
757            // Phase 2 (§4b) — see this method's own docblock above. Native
758            // tags are the SAME $native_tags read above, before translation —
759            // kept distinct from 'tags' (the translated set applied to the
760            // primary post) since Phase 4 (§4d) needs the untranslated names
761            // for the native-language sibling.
762            'native_excerpt' => $source->post_excerpt,
763            'native_body'    => $body_plain,
764            'native_tags'    => $native_tags,
765        ];

861            );
862        }
863
864        if ( ! hash_equals( $stored_token, $token ) ) {
865            return new WP_Error(
866                'agnosis_token_invalid',
867                __( 'Invalid review token.', 'agnosis' ),
868                [ 'status' => 403 ]
869            );
